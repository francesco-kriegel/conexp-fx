249  public final class RowHeaderCell extends Cell<RowHeaderCell, RowHeaderPane> {
250
251    private ImageView view = ImageViewBuilder.create().build();
252
253    private RowHeaderCell(final int row) {
254      super(rowHeaderPane, row, 0, Pos.CENTER_RIGHT, TextAlignment.RIGHT, false, null, false);
255      this.contentPane.get().getChildren().add(view);
256      this.contentPane.get().text.setOpacity(0);
257      if (cellPane.interactive) {
258        final ContextMenu contextMenu = new ContextMenu();
259        final MenuItem editItem = new MenuItem("Edit");
260        final MenuItem removeItem = new MenuItem("Remove");
261        final MenuItem selectItem = new MenuItem("Select");
262        final MenuItem insertItem = new MenuItem("Insert");
263        if (dataset != null && dataset.editable)
264          contextMenu.getItems().addAll(editItem, removeItem, selectItem, insertItem);
265        else
266          contextMenu.getItems().addAll(removeItem, selectItem);
267        insertItem.setOnAction(__ -> {
268          ((FCADataset<String, String>) dataset).addObject(
269              (dataset.context.isHomogen() ? "Element " : "Object ") + IdGenerator.getNextId(dataset),
270              contentCoordinates.get().x());
271          rowHeaderPane.rowOpacityMap.keySet().stream().sorted().filter(i -> i >= contentCoordinates.get().x()).forEach(
272              i -> rowHeaderPane.rowOpacityMap.put(i + 1, rowHeaderPane.rowOpacityMap.remove(i)));
273        });
274        editItem.setOnAction(__ -> {
275          if (MatrixContextWidget.this.dataset.editable) {
276            final G object = context.rowHeads().get(contentCoordinates.get().x());
277            final TextField textField = TextFieldBuilder.create().text((String) object).build();
278            textField.addEventHandler(KeyEvent.KEY_RELEASED, event -> {
279              switch (event.getCode()) {
280              case ENTER:
281                dataset.renameObject(object, (G) textField.getText().trim());
282              case ESCAPE:
283                interactionPane.get().getChildren().remove(textField);
284              }
285            });
286            interactionPane.get().getChildren().add(textField);
287            textField.focusedProperty().addListener(
288                (observable, oldValue, newValue) -> new Timer().schedule(new TimerTask() {
289
290              public final void run() {
291                Platform.runLater(() -> textField.selectAll());
292              }
293            }, 20));
294            textField.requestFocus();
295          } else {
296            System.out.println("no instance of MatrixContextWidget");
297          }
298        });
299        removeItem.setOnAction(__ -> {
300          dataset.removeObject(context.rowHeads().get(contentCoordinates.get().x()));
301          rowHeaderPane.rowOpacityMap.keySet().stream().sorted().filter(i -> i > contentCoordinates.get().x()).forEach(
302              i -> rowHeaderPane.rowOpacityMap.put(i - 1, rowHeaderPane.rowOpacityMap.remove(i)));
303        });
304        selectItem.setOnAction(__ -> select());
305        this.interactionPane.get().addEventHandler(MouseEvent.MOUSE_CLICKED, event -> {
306          switch (event.getButton()) {
307          case PRIMARY:
308            select();
309            break;
310          case SECONDARY:
311            contextMenu.show(interactionPane.getValue(), event.getScreenX(), event.getScreenY());
312          }
313        });
314      }
315      if (dataset != null)
316        this.interactionPane.get().addEventHandler(MouseEvent.MOUSE_ENTERED, event -> {
317          if (contextPane.highlight.get())
318            dataset.conceptGraph.highlight(
319                true,
320                dataset.conceptGraph.highlightRequests.object(context.rowHeads().get(contentCoordinates.get().x())));
321        });
322      if (cellPane.autoSizeRows.get() || cellPane.autoSizeColumns.get())
323        view.layoutBoundsProperty().addListener(new ChangeListener<Bounds>() {
324
325          @Override
326          public void changed(ObservableValue<? extends Bounds> observable, Bounds oldValue, Bounds newValue) {
327            final double decorationWidth =
328                cellPane.decorations.containsKey(context.rowHeads().get(contentCoordinates.get().x()))
329                    ? cellPane.decorations
330                        .get(context.rowHeads().get(contentCoordinates.get().x()))
331                        .layoutBoundsProperty()
332                        .get()
333                        .getWidth()
334                    : 0d;
335            final double width = newValue.getWidth() + decorationWidth;
336            if (width > cellPane.maximalTextWidth.get())
337              cellPane.maximalTextWidth.set(width);
338          }
339        });
340      context.addEventHandler(event -> Platform2.runOnFXThread(RowHeaderCell.this::updateContent), RelationEvent.ROWS);
341      updateContent();
342    }

344    private final void select() {
345      synchronized (rowHeaderPane.rowOpacityMap) {
346        if (rowHeaderPane.rowOpacityMap.containsKey(contentCoordinates.get().x())) {
347          if (dataset != null)
348            dataset.conceptGraph.highlight(false, dataset.conceptGraph.highlightRequests.dehighlight());
349          rowHeaderPane.rowOpacityMap.remove(contentCoordinates.get().x());
350          if (dataset != null)
351            dataset.selectObject(context.rowHeads().get(contentCoordinates.get().x()));
352        } else {
353          if (dataset != null)

355          rowHeaderPane.rowOpacityMap.put(contentCoordinates.get().x(), Constants.HIDE_OPACITY);
356          if (dataset != null)
357            dataset.ignoreObject(context.rowHeads().get(contentCoordinates.get().x()));
358        }
359      }

361//    rowHeaderPane.rowOpacityMap.remove(contentCoordinates.get().x());
362////  final G object = context.getDomain().get(contentCoordinates.get().x());
363////  context.selectObject(object);
364//  } else {

366////  final G object = context.getDomain().get(contentCoordinates.get().x());
367////  context.deselectObject(object);
368//  }
369    }

379              .get()

381        }

383    }

516  public final class ContextCell extends Cell<ContextCell, ContextPane> {
517
518    private ContextCell(final int row, final int column) {
519      super(contextPane, row, column, Pos.CENTER, TextAlignment.CENTER, false, null, false);
520      if (dataset != null)
521        this.interactionPane.get().addEventHandler(MouseEvent.MOUSE_CLICKED, new EventHandler<MouseEvent>() {
522
523          public final void handle(final MouseEvent event) {
524            final G g = context.rowHeads().get(contentCoordinates.get().x());
525            final M m = context.colHeads().get(contentCoordinates.get().y());
526            dataset.conceptGraph.highlight(false, dataset.conceptGraph.highlightRequests.dehighlight());
527            dataset.flip(g, m);
528            dataset.unsavedChanges.set(true);
529          }
530        });
531      else
532        this.interactionPane.get().addEventHandler(MouseEvent.MOUSE_CLICKED, event -> {
533          final G g = context.rowHeads().get(contentCoordinates.get().x());
534          final M m = context.colHeads().get(contentCoordinates.get().y());
535          if (context.contains(g, m))
536            context.remove(g, m);
537          else
538            context.add(g, m);
539        });
540      if (dataset != null)
541        this.interactionPane.get().addEventHandler(MouseEvent.MOUSE_ENTERED, new EventHandler<MouseEvent>() {
542
543          public final void handle(final MouseEvent event) {
544            if (contextPane.highlight.get()) {
545              final G g = context.rowHeads().get(contentCoordinates.get().x());
546              final M m = context.colHeads().get(contentCoordinates.get().y());
547              if (context.selectedAttributes().contains(m))
548                if (textContent.get().equals(Constants.CROSS_CHARACTER))
549                  dataset.conceptGraph.highlight(true, dataset.conceptGraph.highlightRequests.incidence(g, m));
550                else if (textContent.get().equals(Constants.DOWN_ARROW_CHARACTER))
551                  dataset.conceptGraph.highlight(true, dataset.conceptGraph.highlightRequests.downArrow(g, m));
552                else if (textContent.get().equals(Constants.UP_ARROW_CHARACTER))
553                  dataset.conceptGraph.highlight(true, dataset.conceptGraph.highlightRequests.upArrow(g, m));
554                else if (textContent.get().equals(Constants.BOTH_ARROW_CHARACTER))
555                  dataset.conceptGraph.highlight(true, dataset.conceptGraph.highlightRequests.bothArrow(g, m));
556                else if (textContent.get().equals(Constants.NO_CROSS_CHARACTER))
557                  dataset.conceptGraph.highlight(true, dataset.conceptGraph.highlightRequests.nonIncidence(g, m));
558            }
559          }
560        });
561      context.addEventHandler(event -> updateContent(), RelationEvent.ANY);
562//          RelationEvent.ENTRIES_ADDED,
563//          RelationEvent.ENTRIES_REMOVED,
564//          RelationEvent.ALL_CHANGED,
565//          RelationEvent.SELECTION_CHANGED);
566      updateContent();
567    }

569    @SuppressWarnings("incomplete-switch")
570    public final void updateContent() {
571      try {
572        final G g = context.rowHeads().get(contentCoordinates.get().x());
573        final M m = context.colHeads().get(contentCoordinates.get().y());
574        if (context.selectedAttributes().contains(m) && context.selectedObjects().contains(g)) {
575          final Pair<Incidence, Incidence> p = context.selection.getValue(g, m, showArrows.get(), showPaths.get());
576          final Incidence first = p.first();
577          final Incidence second = p.second();
578          ContextCell.this.textContent.set(
579              second != null && first == Incidence.NO_CROSS ? Constants.NO_CROSS_CHARACTER_BOLD : first.toString());
580          switch (first) {
581          case BOTH_ARROW:
582            ContextCell.this.contentPane.get().text.setRotate(-45d);
583            break;
584          case DOWN_ARROW:
585          case UP_ARROW:
586          case CROSS:
587          case NO_CROSS:
588            ContextCell.this.contentPane.get().text.setRotate(0d);
589            break;
590          }
591          if (second == null)
592            ContextCell.this.contentPane.get().text.setFill(Color.BLACK);
593          else
594            switch (second) {
595            case BOTH_PATH:
596              ContextCell.this.contentPane.get().text.setFill(ConceptGraph.COLOR_INTERVAL);
597              break;
598            case DOWN_PATH:
599              ContextCell.this.contentPane.get().text.setFill(ConceptGraph.COLOR_LOWER);
600              break;
601            case UP_PATH:
602              ContextCell.this.contentPane.get().text.setFill(ConceptGraph.COLOR_UPPER);
603              break;
604            }
605        } else {
606          ContextCell.this.textContent.set(context.getValue(g, m, false).first().toString());
607          ContextCell.this.contentPane.get().text.setRotate(0d);
608        }
609      } catch (IndexOutOfBoundsException __) {}
610    }

975  public final void highlight(final Concept<G, M> concept) {
976    final Collection<Integer> domainIndices = context.rowHeads().indicesOf(concept.extent(), true);
977    final Collection<Integer> codomainIndices = context.colHeads().indicesOf(concept.intent(), true);
978    contextPane.highlightConcept.set(true);
979    rowHeaderPane.highlightConcept(Collections2.transform(domainIndices, new Function<Integer, Integer>() {
980
981      public final Integer apply(final Integer index) {
982        for (Entry<Integer, Integer> entry : contextPane.rowMap.get().entrySet())
983          if (entry.getValue().equals(index))
984            return entry.getKey();
985        return index;
986      }
987    }));
988    colHeaderPane.highlightConcept(Collections2.transform(codomainIndices, new Function<Integer, Integer>() {
989
990      public final Integer apply(final Integer index) {
991        for (Entry<Integer, Integer> entry : contextPane.columnMap.get().entrySet())
992          if (entry.getValue().equals(index))
993            return entry.getKey();
994        return index;
995      }
996    }));
997  }
