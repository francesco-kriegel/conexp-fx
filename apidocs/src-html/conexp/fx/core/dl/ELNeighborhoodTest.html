100  private static final void test3() {
101    final Meter<Long> timer = Meter.newNanoStopWatch();
102    final ELConceptDescription C = ELParser.read("exists r.(A and exists r.(A and B) and exists r.(B and C))");
103    System.out.println(C);
104    final AtomicInteger p = new AtomicInteger(0);
105    final BitSetFX ds = new BitSetFX();
106    recurse(C, 0, p::incrementAndGet, d -> {
107      synchronized (ds) {
108        ds.add(d);
109      }
110    });
111    System.out.println("branches: " + p);
112    System.out.println("quasi rank: " + ds);
113    System.out.println("computation time: " + timer.measureAndFormat());
114    timer.reset();
115    System.out.println("rank: " + C.rank());
116    System.out.println("computation time: " + timer.measureAndFormat());
117  }
118
119  private static final void
120      recurse(final ELConceptDescription C, final int d, final Supplier<Integer> pinc, final Consumer<Integer> dadd) {
121    if (C.isTop()) {
122      pinc.get();
123      dadd.accept(d);
124      return;
125    }
126    C.upperNeighborsReduced().parallelStream().forEach(D -> {
127//      D.reduce();
128      recurse(D, d + 1, pinc, dadd);
129    });
130  }

629  private static final boolean
630      containsEquivalent(final Set<ELConceptDescription> Cs, final Set<ELConceptDescription> Ds) {
631    return Cs.parallelStream().allMatch(C -> Ds.parallelStream().anyMatch(C::isEquivalentTo));
632  }
633
634  private static final boolean
635      equalsEquivalent(final Set<ELConceptDescription> Cs, final Set<ELConceptDescription> Ds) {
636    return Stream
637        .<Supplier<Boolean>> of(() -> containsEquivalent(Cs, Ds), () -> containsEquivalent(Ds, Cs))
638        .parallel()
639        .allMatch(Supplier::get);
640  }
