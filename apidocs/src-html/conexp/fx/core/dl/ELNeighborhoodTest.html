092  private static final void test3() {
093    final Meter<Long> timer = Meter.newNanoStopWatch();
094    final ELConceptDescription C = ELParser.read("exists r.(A and exists r.(A and B) and exists r.(B and C))");
095    System.out.println(C);
096    final AtomicInteger p = new AtomicInteger(0);
097    final BitSetFX ds = new BitSetFX();
098    recurse(C, 0, p::incrementAndGet, d -> {
099      synchronized (ds) {
100        ds.add(d);
101      }
102    });
103    System.out.println("branches: " + p);
104    System.out.println("quasi rank: " + ds);
105    System.out.println("computation time: " + timer.measureAndFormat());
106    timer.reset();
107    System.out.println("rank: " + C.rank());
108    System.out.println("computation time: " + timer.measureAndFormat());
109  }
110
111  private static final void
112      recurse(final ELConceptDescription C, final int d, final Supplier<Integer> pinc, final Consumer<Integer> dadd) {
113    if (C.isTop()) {
114      pinc.get();
115      dadd.accept(d);
116      return;
117    }
118    C.upperNeighborsReduced().parallelStream().forEach(D -> {
119//      D.reduce();
120      recurse(D, d + 1, pinc, dadd);
121    });
122  }

431  private static final boolean
432      containsEquivalent(final Set<ELConceptDescription> Cs, final Set<ELConceptDescription> Ds) {
433    return Cs.parallelStream().allMatch(C -> Ds.parallelStream().anyMatch(C::isEquivalentTo));
434  }
435
436  private static final boolean
437      equalsEquivalent(final Set<ELConceptDescription> Cs, final Set<ELConceptDescription> Ds) {
438    return Stream
439        .<Supplier<Boolean>> of(() -> containsEquivalent(Cs, Ds), () -> containsEquivalent(Ds, Cs))
440        .parallel()
441        .allMatch(Supplier::get);
442  }
