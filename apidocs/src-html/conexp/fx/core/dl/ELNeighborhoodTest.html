106  private static final void
107      recurse(final ELConceptDescription C, final int d, final Supplier<Integer> pinc, final Consumer<Integer> dadd) {
108    if (C.isTop()) {
109      pinc.get();
110      dadd.accept(d);
111      return;
112    }
113    C.upperNeighborsReduced().parallelStream().forEach(D -> {
114//      D.reduce();
115      recurse(D, d + 1, pinc, dadd);
116    });
117  }

220  private static final boolean
221      containsEquivalent(final Set<ELConceptDescription> Cs, final Set<ELConceptDescription> Ds) {
222    return Cs.parallelStream().allMatch(C -> Ds.parallelStream().anyMatch(C::isEquivalentTo));
223  }
224
225  private static final boolean
226      equalsEquivalent(final Set<ELConceptDescription> Cs, final Set<ELConceptDescription> Ds) {
227    return Stream
228        .<Supplier<Boolean>> of(() -> containsEquivalent(Cs, Ds), () -> containsEquivalent(Ds, Cs))
229        .parallel()
230        .allMatch(Supplier::get);
231  }
