025import java.util.HashMap;
026import java.util.HashSet;
027import java.util.Map;
028import java.util.Set;
029import java.util.stream.Collectors;
030import java.util.stream.Stream;
031
032import org.ujmp.core.Matrix;
033import org.ujmp.core.enums.ValueType;
034
035import conexp.fx.core.collections.Pair;
036
037public class Matrix3D<G, M, W> {
038
039  protected final Matrix        mat;
040  private int                   xs = 0;
041  private int                   ys = 0;
042  private int                   zs = 0;
043  private final Map<G, Integer> x1;
044  private final Map<M, Integer> y1;
045  private final Map<W, Integer> z1;
046  private final Map<Integer, G> x2;
047  private final Map<Integer, M> y2;
048  private final Map<Integer, W> z2;
049
050  public Matrix3D(final int x, final int y, final int z) {
051    super();
052    mat = Matrix.Factory.zeros(ValueType.BOOLEAN, x, y, z);
053    x1 = new HashMap<G, Integer>(x);
054    y1 = new HashMap<M, Integer>(y);
055    z1 = new HashMap<W, Integer>(z);
056    x2 = new HashMap<Integer, G>(x);
057    y2 = new HashMap<Integer, M>(y);
058    z2 = new HashMap<Integer, W>(z);
059  }
060
061  public final void addG(final G g) {
062    if (x1.keySet().contains(g))
063      return;
064    x1.put(g, xs);
065    x2.put(xs, g);
066    xs++;
067  }
068
069  public final void addM(final M m) {
070    if (y1.keySet().contains(m))
071      return;
072    y1.put(m, ys);
073    y2.put(ys, m);
074    ys++;
075  }
076
077  public final void addW(final W w) {
078    if (z1.keySet().contains(w))
079      return;
080    z1.put(w, zs);
081    z2.put(zs, w);
082    zs++;
083  }
084
085  public final Set<G> getGs() {
086    return x1.keySet();
087  }
088
089  public final Set<M> getMs() {
090    return y1.keySet();
091  }
092
093  public final Set<W> getWs() {
094    return z1.keySet();
095  }
096
097  public final void add(final G g, final M m, final W w) throws NullPointerException {
098    mat.setAsBoolean(true, x1.get(g), y1.get(m), z1.get(w));
099  }
100
101  public final boolean get(final G g, final M m, final W w) throws NullPointerException {
102    return mat.getAsBoolean(x1.get(g), y1.get(m), z1.get(w));
103  }
104
105  public final Stream<G> row(final M m, final W w) {
106    final int y = y1.get(m);
107    final int z = z1.get(w);
108    return x1.keySet().stream().filter(g -> mat.getAsBoolean(x1.get(g), y, z));
109  }
110
111  public final Stream<M> col(final G g, final W w) {
112    final int x = x1.get(g);
113    final int z = z1.get(w);
114    return y1.keySet().stream().filter(m -> mat.getAsBoolean(x, y1.get(m), z));
115  }
116
117  public final Stream<W> cut(final G g, final M m) {
118    final int x = x1.get(g);
119    final int y = y1.get(m);
120    return z1.keySet().stream().filter(w -> mat.getAsBoolean(x, y, z1.get(w)));
121  }
122
123  public final Stream<W> cut(final Stream<G> gs, final M m) {
124    final Set<G> _gs = gs.collect(Collectors.toSet());
125    final int y = y1.get(m);
126    return z1.keySet().stream().filter(w -> _gs.stream().allMatch(g -> mat.getAsBoolean(x1.get(g), y, z1.get(w))));
127  }
128
129  public final Set<Pair<G, W>> getPairsGW(final M m) {
130    final Set<Pair<G, W>> pairs = new HashSet<Pair<G, W>>();
131    for (G g : getGs())
132      for (W w : getWs())
133        if (get(g, m, w))
134          pairs.add(new Pair<G, W>(g, w));
135    return pairs;
136  }
137
138}
