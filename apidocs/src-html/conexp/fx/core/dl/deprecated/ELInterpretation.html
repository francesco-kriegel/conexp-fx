089                existentialRestriction -> getRoleSuccessorStream(existentialRestriction.getKey(), individual)

231  @Override
232  protected final SetList<ELConceptDescription> getAttributeSetForInducedContext(
233      final int roleDepth,
234      final int maxCardinality,
235      final Constructor... constructors) {
236    final Set<ELConceptDescription> mmscs = getAllMostSpecificConcepts(roleDepth - 1, 0);
237    final SetList<ELConceptDescription> _codomain = new HashSetArrayList<ELConceptDescription>();
238    _codomain.add(ELConceptDescription.bot());
239    _codomain.addAll(Collections2.transform(signature.getConceptNames(), ELConceptDescription::conceptName));
240    for (IRI roleName : signature.getRoleNames())
241      _codomain.addAll(Collections2.transform(mmscs, mmsc -> {
242        return ELConceptDescription.existentialRestriction(roleName, mmsc);
243      }));
244    return _codomain;
245  };

247  @Override
248  protected final Set<Implication<IRI, ELConceptDescription>>
249      getBackgroundImplications(final Context<IRI, ELConceptDescription> inducedContext, final ELTBox backgroundTBox) {
250    final BiPredicate<ELConceptDescription, ELConceptDescription> subsumptionTest;
251    if (backgroundTBox == null)
252      subsumptionTest = (concept1, concept2) -> ELReasoner.isSubsumedBy(concept1, concept2);
253    else
254      subsumptionTest = (concept1, concept2) -> ELReasoner.isSubsumedBy(concept1, concept2, backgroundTBox);
255    final Set<Implication<IRI, ELConceptDescription>> backgroundImplications =
256        new HashSet<Implication<IRI, ELConceptDescription>>();
257    for (ELConceptDescription concept1 : inducedContext.colHeads())
258      for (ELConceptDescription concept2 : inducedContext.colHeads())
259        if (!concept1.equals(concept2))
260          if (subsumptionTest.test(concept1, concept2))
261            backgroundImplications
262                .add(
263                    new Implication<IRI, ELConceptDescription>(
264                        Collections.singleton(concept1),
265                        Collections.singleton(concept2),
266                        Collections.emptySet()));
267    return backgroundImplications;
268  }

270  @Override
271  public final ELTBox computeTBoxBase(
272      final int roleDepth,
273      final int maxCardinality,
274      final ELTBox backgroundTBox,
275      final Constructor... constructors) {
276    checkRoleDepth(roleDepth);
277    final ELTBox tbox = new ELTBox();
278    final Context<IRI, ELConceptDescription> inducedContext = getInducedContext(domain, roleDepth, maxCardinality);
279    final Set<Implication<IRI, ELConceptDescription>> backgroundImplications =
280        getBackgroundImplications(inducedContext, backgroundTBox);
281    final ResultC<IRI, ELConceptDescription> result =
282        NextClosures1C.computeWithBackgroundImplications(inducedContext, backgroundImplications, false);
283    for (Entry<Set<ELConceptDescription>, Set<ELConceptDescription>> entry : result.implications.entrySet())
284      tbox
285          .getGCIs()
286          .add(
287              new ELConceptInclusion(
288                  ELConceptDescription.conjunction(entry.getKey()).clone().reduce(),
289                  ELConceptDescription.conjunction(entry.getValue()).clone().reduce()));
290    return tbox;
291  }

293  public static final ELInterpretation fromTriples(final File rdfFile, final IRI baseIRI, final String isaRole) {
294    final ELInterpretation i = new ELInterpretation(baseIRI);
295    final Multimap<String, String> concepts = HashMultimap.create();
296    final Multimap<String, Pair<String, String>> roles = HashMultimap.create();
297    try {
298      final BufferedReader reader = new BufferedReader(new FileReader(rdfFile));
299      final Iterator<String> it = reader.lines().iterator();
300      while (it.hasNext()) {
301        String next = it.next();
302        while (next.contains("  "))
303          next = next.replace("  ", " ");
304        next = next.replace("<", "").replace(">", "");
305        final String[] triple = next.split(" ");
306        if (triple.length > 2)
307          if (triple[1].contains(isaRole)) {
308            concepts.put(triple[0], triple[2]);
309          } else {
310            roles.put(triple[1], new Pair<String, String>(triple[0], triple[2]));
311          }
312      }
313      reader.close();
314    } catch (IOException e) {
315      throw new UncheckedIOException(e);
316    }
317    for (Entry<String, String> entry : concepts.entries()) {
318      final IRI c = IRI.create(entry.getKey());
319      final IRI d = IRI.create(entry.getValue());
320      i.getSignature().getConceptNames().add(d);
321      i.getDomain().add(c);
322      i.addConceptNameAssertion(d, c);
323    }
324    for (Entry<String, Pair<String, String>> entry : roles.entries()) {
325      final IRI r = IRI.create(entry.getKey());
326      final IRI d = IRI.create(entry.getValue().x());
327      final IRI e = IRI.create(entry.getValue().y());
328      i.getSignature().getRoleNames().add(r);
329      i.getDomain().add(d);
330      i.getDomain().add(e);
331      i.addRoleNameAssertion(r, d, e);
332    }
333    return i;
334  }

336  public static final ELInterpretation fromTriples(final List<IRI[]> triples, final IRI baseIRI, final IRI isaRole) {
337    final ELInterpretation i = new ELInterpretation(baseIRI);
338    for (IRI[] triple : triples) {
339      if (triple[1].equals(isaRole)) {
340        i.getSignature().getConceptNames().add(triple[2]);
341        i.getDomain().add(triple[0]);
342        i.addConceptNameAssertion(triple[2], triple[0]);
343      } else {
344        i.getSignature().getRoleNames().add(triple[1]);
345        i.getDomain().add(triple[0]);
346        i.getDomain().add(triple[2]);
347        i.addRoleNameAssertion(triple[1], triple[0], triple[2]);
348      }
349    }
350    return i;
351  }
