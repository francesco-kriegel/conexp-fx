100  public final boolean isInstanceOf(final IRI individual, final OWLClassExpression conceptExpression) {
101    if (conceptExpression.isOWLNothing())
102      return false;
103    if (conceptExpression.isOWLThing())
104      return true;
105    if (conceptExpression instanceof OWLClass)
106      return getConceptNameExtension(((OWLClass) conceptExpression).getIRI()).contains(individual);
107    if (conceptExpression instanceof OWLObjectIntersectionOf)
108      return ((OWLObjectIntersectionOf) conceptExpression).asConjunctSet().parallelStream().allMatch(
109          c -> isInstanceOf(individual, c));
110    if (conceptExpression instanceof OWLObjectSomeValuesFrom) {
111      final OWLObjectSomeValuesFrom existentialRestriction = (OWLObjectSomeValuesFrom) conceptExpression;
112      return getRoleSuccessorStream(((OWLObjectProperty) existentialRestriction.getProperty()).getIRI(), individual)
113          .anyMatch(succ -> isInstanceOf(succ, existentialRestriction.getFiller()));
114    }
115    if (conceptExpression instanceof OWLObjectAllValuesFrom) {
116      final OWLObjectAllValuesFrom valueRestriction = (OWLObjectAllValuesFrom) conceptExpression;
117      return getRoleSuccessorStream(((OWLObjectProperty) valueRestriction.getProperty()).getIRI(), individual)
118          .allMatch(succ -> isInstanceOf(succ, valueRestriction.getFiller()));
119    }
120    if (conceptExpression instanceof OWLObjectHasSelf) {
121      final OWLObjectHasSelf existentialSelfRestriction = (OWLObjectHasSelf) conceptExpression;
122      return getRoleNameExtension(((OWLObjectProperty) existentialSelfRestriction.getProperty()).getIRI())
123          .contains(Pair.of(individual, individual));
124    }
125    if (conceptExpression instanceof OWLObjectMinCardinality) {
126      final OWLObjectMinCardinality qualifiedNumberRestriction = (OWLObjectMinCardinality) conceptExpression;
127      return getRoleSuccessorStream(((OWLObjectProperty) qualifiedNumberRestriction.getProperty()).getIRI(), individual)
128          .filter(succ -> isInstanceOf(succ, qualifiedNumberRestriction.getFiller()))
129          .skip(qualifiedNumberRestriction.getCardinality() - 1)
130          .findAny()
131          .isPresent();
132    }
133    if (conceptExpression instanceof OWLObjectMaxCardinality) {
134      final OWLObjectMaxCardinality qualifiedNumberRestriction = (OWLObjectMaxCardinality) conceptExpression;
135      return !getRoleSuccessorStream(
136          ((OWLObjectProperty) qualifiedNumberRestriction.getProperty()).getIRI(),
137          individual)
138              .filter(succ -> isInstanceOf(succ, qualifiedNumberRestriction.getFiller()))
139              .skip(qualifiedNumberRestriction.getCardinality())
140              .findAny()
141              .isPresent();
142    }
143    if (conceptExpression instanceof OWLObjectExactCardinality) {
144      final OWLObjectExactCardinality qualifiedNumberRestriction = (OWLObjectExactCardinality) conceptExpression;
145      return getRoleSuccessorStream(((OWLObjectProperty) qualifiedNumberRestriction.getProperty()).getIRI(), individual)
146          .filter(succ -> isInstanceOf(succ, qualifiedNumberRestriction.getFiller()))
147          .skip(qualifiedNumberRestriction.getCardinality() - 1)
148          .findAny()
149          .isPresent()
150          && !getRoleSuccessorStream(
151              ((OWLObjectProperty) qualifiedNumberRestriction.getProperty()).getIRI(),
152              individual)
153                  .filter(succ -> isInstanceOf(succ, qualifiedNumberRestriction.getFiller()))
154                  .skip(qualifiedNumberRestriction.getCardinality())
155                  .findAny()
156                  .isPresent();
157    }
158    if (conceptExpression instanceof OWLObjectComplementOf) {
159      final OWLObjectComplementOf primitiveNegation = (OWLObjectComplementOf) conceptExpression;
160      if (primitiveNegation.isClassExpressionLiteral())
161        return !isInstanceOf(individual, (OWLClass) primitiveNegation.getOperand());
162    }
163    throw new ELSyntaxException();
164  }

218  public OWLClassExpression getMostSpecificConceptALQ(
219      final Set<IRI> individuals,
220      final int roleDepth,
221      final int maxCardinality,
222      final Constructor... constructors) {
223    checkRoleDepth(roleDepth);
224    if (individuals.isEmpty())
225      return df.getOWLNothing();
226    final Set<OWLClassExpression> conjuncts = new HashSet<OWLClassExpression>();
227    for (IRI conceptName : conceptNameExtensions.keySet())
228      if (getConceptNameExtension(conceptName).containsAll(individuals))
229        conjuncts.add(new OWLClassImpl(conceptName));
230      else if (Arrays.asList(constructors).contains(Constructor.PRIMITIVE_NEGATION)
231          && Collections2.filter(getDomain(), d -> !getConceptNameExtension(conceptName).contains(d)).containsAll(
232              individuals))
233        conjuncts.add(new OWLObjectComplementOfImpl(new OWLClassImpl(conceptName)));
234    if (roleDepth > 0) {
235      for (IRI roleName : signature.getRoleNames()) {
236        final OWLObjectProperty property = df.getOWLObjectProperty(roleName);
237        if (Arrays.asList(constructors).contains(Constructor.EXISTENTIAL_SELF_RESTRICTION)) {
238          if (individuals.parallelStream().allMatch(
239              individual -> getRoleNameExtension(roleName).contains(Pair.of(individual, individual))))
240            conjuncts.add(df.getOWLObjectHasSelf(property));
241        }
242        if (Arrays.asList(constructors).contains(Constructor.EXISTENTIAL_RESTRICTION)) {
243          for (Set<IRI> successors : getSuccessorSetsER(individuals, roleName))
244            conjuncts.add(
245                df.getOWLObjectSomeValuesFrom(
246                    property,
247                    getMostSpecificConceptALQ(successors, roleDepth - 1, maxCardinality, constructors)));
248        }
249        if (Arrays.asList(constructors).contains(Constructor.VALUE_RESTRICTION)) {
250//        for (Set<IRI> successors : getSuccessorSetsVR(individuals, roleName))
251          conjuncts.add(
252              df.getOWLObjectAllValuesFrom(
253                  property,
254                  getMostSpecificConceptALQ(
255                      getAllSuccessors(individuals, roleName),
256                      roleDepth - 1,
257                      maxCardinality,
258                      constructors)));
259        }
260        if (Arrays.asList(constructors).contains(Constructor.UNQUALIFIED_AT_MOST_RESTRICTION)) {
261          for (int cardinality = 1; cardinality < maxCardinality; cardinality++)
262            for (Set<IRI> successors : getSuccessorSetsQLR(individuals, roleName, cardinality))
263              conjuncts.add(
264                  df.getOWLObjectMaxCardinality(
265                      cardinality,
266                      property,
267                      getMostSpecificConceptALQ(successors, roleDepth - 1, maxCardinality, constructors)));
268        }
269        if (Arrays.asList(constructors).contains(Constructor.QUALIFIED_AT_LEAST_RESTRICTION)) {
270          for (int cardinality = 2; cardinality < maxCardinality; cardinality++)
271            for (Set<IRI> successors : getSuccessorSetsQGR(individuals, roleName, cardinality))
272              conjuncts.add(
273                  df.getOWLObjectMinCardinality(
274                      cardinality,
275                      property,
276                      getMostSpecificConceptALQ(successors, roleDepth - 1, maxCardinality, constructors)));
277        }
278        // TODO
279      }
280    }
281    if (conjuncts.isEmpty())
282      return df.getOWLThing();
283    if (conjuncts.size() == 1)
284      return conjuncts.iterator().next();
285    return new OWLObjectIntersectionOfImpl(conjuncts);
286  }

302    return filterMinimal(

313  protected final Set<Set<IRI>> getSuccessorSetsER2(final Set<IRI> individuals, final IRI roleName) {
314    // this method should not filter the powerset, which is a costly operation, but instead create the minimal
315    // successors directly
316    Set<Set<IRI>> successorSets = Collections.emptySet();
317    final Iterator<IRI> it = individuals.iterator();
318    if (it.hasNext()) {
319      successorSets = firstSuccessorSetsER(it.next(), roleName);
320    }
321    while (it.hasNext()) {
322      successorSets = extendSuccessorSetsER(successorSets, it.next(), roleName);
323    }
324    return filterMinimal(successorSets);
325  }

332  private final Set<Set<IRI>>
333      extendSuccessorSetsER(final Set<Set<IRI>> successorSets, final IRI individual, final IRI roleName) {
334    final Set<Set<IRI>> extendedSets = new HashSet<Set<IRI>>();
335    final Set<Set<IRI>> extendedSetsP = Collections.synchronizedSet(extendedSets);
336    successorSets.parallelStream().forEach(successorSet -> {
337      getRoleSuccessorStream(roleName, individual).forEach(successor -> {
338        final Set<IRI> extendedSet = new HashSet<IRI>();
339        extendedSet.addAll(successorSet);
340        extendedSet.add(successor);
341        extendedSetsP.add(extendedSet);
342      });
343    });
344    return extendedSets;
345  }

496  @Override
497  protected final SetList<OWLClassExpression> getAttributeSetForInducedContext(
498      final int roleDepth,
499      final int maxCardinality,
500      final Constructor... constructors) {
501    final Set<OWLClassExpression> mmscs = roleDepth > 0
502        ? getAllMostSpecificConcepts(roleDepth - 1, maxCardinality, constructors) : Collections.emptySet();
503    final SetList<OWLClassExpression> _codomain = new HashSetArrayList<OWLClassExpression>();
504    _codomain.add(df.getOWLNothing());
505    _codomain.addAll(Collections2.transform(signature.getConceptNames(), df::getOWLClass));
506    if (Arrays.asList(constructors).contains(Constructor.PRIMITIVE_NEGATION))
507      _codomain.addAll(
508          Collections2.transform(
509              signature.getConceptNames(),
510              conceptName -> df.getOWLObjectComplementOf(df.getOWLClass(conceptName))));
511    for (IRI roleName : signature.getRoleNames())
512      for (Constructor c : constructors)
513        switch (c) {
514        case EXISTENTIAL_RESTRICTION:
515          _codomain.addAll(Collections2.transform(mmscs, mmsc -> {
516            return df.getOWLObjectSomeValuesFrom(df.getOWLObjectProperty(roleName), mmsc);
517          }));
518          break;
519        case VALUE_RESTRICTION:
520          _codomain.addAll(Collections2.transform(mmscs, mmsc -> {
521            return df.getOWLObjectAllValuesFrom(df.getOWLObjectProperty(roleName), mmsc);
522          }));
523          break;
524        case QUALIFIED_AT_LEAST_RESTRICTION:
525          Stream
526              .iterate(0, n -> n + 1)
527              .limit(maxCardinality)
528              .skip(2)
529              .map(
530                  cardinality -> Collections2
531                      .transform(
532                          mmscs,
533                          mmsc -> df.getOWLObjectMinCardinality(cardinality, df.getOWLObjectProperty(roleName), mmsc))
534                      .stream())
535              .reduce(Stream::concat)
536              .ifPresent(stream -> _codomain.addAll(stream.collect(Collectors.toSet())));
537          break;
538        case UNQUALIFIED_AT_MOST_RESTRICTION:
539          Stream
540              .iterate(0, n -> n + 1)
541              .limit(maxCardinality)

565  @Override
566  protected final Set<Implication<IRI, OWLClassExpression>> getBackgroundImplications(
567      final Context<IRI, OWLClassExpression> inducedContext,
568      final OWLOntology backgroundOntology) {
569    final BiPredicate<OWLClassExpression, OWLClassExpression> subsumptionTest;
570    if (backgroundOntology == null)
571      subsumptionTest = (concept1, concept2) -> ELReasoner.isSubsumedBy(concept1, concept2);
572    else
573      subsumptionTest = (concept1, concept2) -> ELReasoner.isSubsumedBy(concept1, concept2, backgroundOntology);
574    final Set<Implication<IRI, OWLClassExpression>> backgroundImplications =
575        new HashSet<Implication<IRI, OWLClassExpression>>();
576    for (OWLClassExpression concept1 : inducedContext.colHeads())
577      for (OWLClassExpression concept2 : inducedContext.colHeads())
578        if (!concept1.equals(concept2)) {
579          if (subsumptionTest.test(concept1, concept2))
580            backgroundImplications.add(
581                new Implication<IRI, OWLClassExpression>(
582                    Collections.singleton(concept1),
583                    Collections.singleton(concept2),
584                    Collections.emptySet()));
585        }
586    return backgroundImplications;
587  }
