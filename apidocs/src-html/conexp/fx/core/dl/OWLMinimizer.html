025import java.util.Set;
026import java.util.function.BiPredicate;
027import java.util.stream.Collectors;
028
029import org.semanticweb.HermiT.Reasoner;
030import org.semanticweb.owlapi.apibinding.OWLManager;
031import org.semanticweb.owlapi.model.OWLClassExpression;
032import org.semanticweb.owlapi.model.OWLObjectAllValuesFrom;
033import org.semanticweb.owlapi.model.OWLObjectComplementOf;
034import org.semanticweb.owlapi.model.OWLObjectExactCardinality;
035import org.semanticweb.owlapi.model.OWLObjectIntersectionOf;
036import org.semanticweb.owlapi.model.OWLObjectMaxCardinality;
037import org.semanticweb.owlapi.model.OWLObjectMinCardinality;
038import org.semanticweb.owlapi.model.OWLObjectSomeValuesFrom;
039import org.semanticweb.owlapi.model.OWLOntologyCreationException;
040
041public class OWLMinimizer {
042
043  public static final Reasoner r = initializeReasoner();
044
045  private static final Reasoner initializeReasoner() {
046    try {
047      return new Reasoner(OWLManager.createOWLOntologyManager().createOntology());
048    } catch (OWLOntologyCreationException e) {
049      e.printStackTrace();
050      return null;
051    }
052  }
053
054  public static final boolean isSubsumedBy(final OWLClassExpression c1, final OWLClassExpression c2) {
055    return r.isEntailed(OWLManager.getOWLDataFactory().getOWLSubClassOfAxiom(
056        c1,
057        c2));
058  }
059
060  public static final boolean subsumes(final OWLClassExpression c1, final OWLClassExpression c2) {
061    return isSubsumedBy(
062        c2,
063        c1);
064  }
065
066  public static final <T> Set<T> filterMinimal(final Set<T> elements, BiPredicate<T, T> predicate) {
067    return elements.stream().filter(
068        x -> !elements.stream().anyMatch(
069            y -> !x.equals(y) && predicate.test(
070                x,
071                y))).collect(
072        Collectors.toSet());
073  }
074
075  public static final <T> Set<T> filterMinimalParallel(final Set<T> elements, BiPredicate<T, T> predicate) {
076    return elements.parallelStream().filter(
077        x -> !elements.parallelStream().anyMatch(
078            y -> !x.equals(y) && predicate.test(
079                x,
080                y))).collect(
081        Collectors.toSet());
082  }
083
084  public static final OWLClassExpression minimizeConjunction(final OWLClassExpression classExpression) {
085    if (classExpression instanceof OWLObjectIntersectionOf) {
086      final OWLObjectIntersectionOf c = (OWLObjectIntersectionOf) classExpression;
087      return OWLManager.getOWLDataFactory().getOWLObjectIntersectionOf(
088          filterMinimal(
089              c.getOperands(),
090              OWLMinimizer::subsumes));
091    } else if (classExpression instanceof OWLObjectComplementOf) {
092      final OWLObjectComplementOf c = (OWLObjectComplementOf) classExpression;
093      return OWLManager.getOWLDataFactory().getOWLObjectComplementOf(
094          minimizeConjunction(c.getOperand()));
095    } else if (classExpression instanceof OWLObjectSomeValuesFrom) {
096      final OWLObjectSomeValuesFrom c = (OWLObjectSomeValuesFrom) classExpression;
097      return OWLManager.getOWLDataFactory().getOWLObjectSomeValuesFrom(
098          c.getProperty(),
099          minimizeConjunction(c.getFiller()));
100    } else if (classExpression instanceof OWLObjectAllValuesFrom) {
101      final OWLObjectAllValuesFrom c = (OWLObjectAllValuesFrom) classExpression;
102      return OWLManager.getOWLDataFactory().getOWLObjectSomeValuesFrom(
103          c.getProperty(),
104          minimizeConjunction(c.getFiller()));
105    } else if (classExpression instanceof OWLObjectMinCardinality) {
106      final OWLObjectMinCardinality c = (OWLObjectMinCardinality) classExpression;
107      return OWLManager.getOWLDataFactory().getOWLObjectMinCardinality(
108          c.getCardinality(),
109          c.getProperty(),
110          minimizeConjunction(c.getFiller()));
111    } else if (classExpression instanceof OWLObjectMaxCardinality) {
112      final OWLObjectMaxCardinality c = (OWLObjectMaxCardinality) classExpression;
113      return OWLManager.getOWLDataFactory().getOWLObjectMaxCardinality(
114          c.getCardinality(),
115          c.getProperty(),
116          minimizeConjunction(c.getFiller()));
117    } else if (classExpression instanceof OWLObjectExactCardinality) {
118      final OWLObjectExactCardinality c = (OWLObjectExactCardinality) classExpression;
119      return OWLManager.getOWLDataFactory().getOWLObjectExactCardinality(
120          c.getCardinality(),
121          c.getProperty(),
122          minimizeConjunction(c.getFiller()));
123    }
124    return classExpression;
125  }
126}
