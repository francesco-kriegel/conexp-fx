136   *          Creates a new EL normal form from an OWLClassExpression.

138  public ELConceptDescription(final OWLClassExpression concept) {
139    super();
140    this.conceptNames = new HashSet<>();
141    this.existentialRestrictions = HashMultimap.create();
142    if (concept.isOWLThing())
143      return;
144    if (concept.isOWLNothing()) {
145      conceptNames.add(df.getOWLNothing().getIRI());
146      return;
147    }
148    if (concept instanceof OWLClass) {
149      this.conceptNames.add(((OWLClass) concept).getIRI());
150      return;
151    }
152    if (concept instanceof OWLObjectSomeValuesFrom) {
153      final OWLObjectSomeValuesFrom existentialRestriction = (OWLObjectSomeValuesFrom) concept;
154      this.existentialRestrictions.put(
155          ((OWLObjectProperty) existentialRestriction.getProperty()).getIRI(),
156          new ELConceptDescription(existentialRestriction.getFiller()));
157      return;
158    }
159    if (concept instanceof OWLObjectIntersectionOf) {
160      final OWLObjectIntersectionOf conjunction = (OWLObjectIntersectionOf) concept;
161      for (OWLClassExpression conjunct : conjunction.asConjunctSet())
162        if (conjunct instanceof OWLClass)
163          this.conceptNames.add(((OWLClass) conjunct).getIRI());
164        else if (conjunct instanceof OWLObjectSomeValuesFrom)
165          this.existentialRestrictions.put(
166              ((OWLObjectProperty) ((OWLObjectSomeValuesFrom) conjunct).getProperty()).getIRI(),
167              new ELConceptDescription(((OWLObjectSomeValuesFrom) conjunct).getFiller()));
168        else
169          throw new ELSyntaxException();
170      return;
171    }
172    throw new ELSyntaxException();
173  }

211  public final OWLClassExpression toOWLClassExpression() {
212    if (isTop())
213      return df.getOWLThing();
214    if (isBot())
215      return df.getOWLNothing();
216    if (conceptNames.size() == 1 && existentialRestrictions.isEmpty())
217      return df.getOWLClass(conceptNames.iterator().next());
218    if (conceptNames.isEmpty() && existentialRestrictions.size() == 1) {
219      final Entry<IRI, ELConceptDescription> existentialRestriction =
220          existentialRestrictions.entries().iterator().next();
221      return df.getOWLObjectSomeValuesFrom(
222          df.getOWLObjectProperty(existentialRestriction.getKey()),
223          existentialRestriction.getValue().toOWLClassExpression());
224    }
225    final Set<OWLClassExpression> conjuncts = new HashSet<>();
226    for (IRI conceptName : conceptNames)
227      conjuncts.add(df.getOWLClass(conceptName));
228    for (Entry<IRI, ELConceptDescription> existentialRestriction : existentialRestrictions.entries())
229      conjuncts.add(
230          df.getOWLObjectSomeValuesFrom(
231              df.getOWLObjectProperty(existentialRestriction.getKey()),
232              existentialRestriction.getValue().toOWLClassExpression()));
233    return df.getOWLObjectIntersectionOf(conjuncts);
234  }

257  public final ELConceptDescription reduce() {
258    for (Entry<IRI, ELConceptDescription> er : existentialRestrictions.entries())
259      er.getValue().reduce();
260    final Function<Entry<IRI, ELConceptDescription>, Stream<Pair<Entry<IRI, ELConceptDescription>, Entry<IRI, ELConceptDescription>>>> f =
261        er1 -> existentialRestrictions
262            .entries()
263            .parallelStream()
264            .filter(er2 -> er1.getKey().equals(er2.getKey()))
265            .filter(er2 -> !er1.equals(er2))
266            .map(er2 -> Pair.of(er1, er2));
267    final Set<Entry<IRI, ELConceptDescription>> superfluousERs = existentialRestrictions
268        .entries()
269        .parallelStream()
270        .map(f)
271        .reduce(Stream::concat)
272        .orElseGet(Stream::empty)
273        .filter(p -> p.first().getValue().isSubsumedBy(p.second().getValue()))
274        .map(Pair::second)
275        .collect(Collectors.toSet());
276    superfluousERs.forEach(x -> existentialRestrictions.remove(x.getKey(), x.getValue()));
277    return this;
278  }

280  public final int roleDepth() {
281    if (existentialRestrictions.isEmpty())
282      return 0;
283    return 1 + existentialRestrictions
284        .entries()
285        .parallelStream()
286        .map(Entry::getValue)
287        .map(ELConceptDescription::roleDepth)
288        .max(Integer::compare)
289        .orElse(0);
290  }

315  private final ELConceptDescription oneUpperNeighbor() {
316    if (!this.conceptNames.isEmpty()) {
317      final ELConceptDescription upperNeighbor = this.clone();
318      final Iterator<IRI> it = upperNeighbor.conceptNames.iterator();
319      it.next();
320      it.remove();
321      return upperNeighbor;
322    } else if (!this.existentialRestrictions.isEmpty()) {
323      final ELConceptDescription upperNeighbor = this.clone();
324      final Iterator<Entry<IRI, ELConceptDescription>> it = upperNeighbor.existentialRestrictions.entries().iterator();
325      final Entry<IRI, ELConceptDescription> ER = it.next();
326      it.remove();
327      ER
328          .getValue()
329          .upperNeighborsReduced()
330          .parallelStream()
331          .filter(
332              uER -> upperNeighbor.existentialRestrictions
333                  .entries()
334                  .parallelStream()
335                  .filter(otherER -> !otherER.equals(ER))
336                  .filter(otherER -> ER.getKey().equals(otherER.getKey()))
337                  .map(Entry::getValue)
338                  .noneMatch(uER::subsumes))
339          .map(uER -> Pair.of(ER.getKey(), uER))
340          .sequential()
341          .forEach(p -> upperNeighbor.existentialRestrictions.put(p.first(), p.second()));
342      return upperNeighbor;
343    } else
344      return null;
345  }

443  private static final int recurseUnreducedRank4(final Collection<ELConceptDescription> Ds) {
444    if (Ds.isEmpty())
445      return 0;
446    else if (Ds.size() == 1)
447      return 1 + recurseUnreducedRank4(Ds.iterator().next().upperNeighborsReduced());
448    else
449      return IntStream
450          .range(1, Ds.size() + 1)
451          .boxed()
452          .map(
453              i -> (int) (Math.pow(-1, i + 1) * Sets
454                  .combinations(Sets.newHashSet(Ds), i)
455                  .parallelStream()
456                  .map(ELLeastCommonSubsumer::lcs)
457                  .map(ELConceptDescription::reduce)
458                  .collect(Collectors.toSet())
459                  .parallelStream()
460                  .map(Collections::singleton)
461                  .collect(Collectors.summingInt(ELConceptDescription::recurseUnreducedRank4))))
462          .collect(Collectors.summingInt(Integer::valueOf));
463  }

465  public final int distanceTo(final ELConceptDescription other) {
466    final List<Integer> x = Stream
467        .<Supplier<Integer>> of(
468            () -> ELConceptDescription.conjunction(this, other).rank(),
469            () -> ELLeastCommonSubsumer.lcs(this, other).rank())
470        .parallel()
471        .map(Supplier::get)
472        .collect(Collectors.toList());
473    return x.get(0) - x.get(1);
474  }

477    final ELConceptDescription lcs = ELLeastCommonSubsumer.lcs(this, other);

485  private final ELConceptDescription oneUpperNeighborBelow(final ELConceptDescription other) {
486    return Stream.concat(this.conceptNames.parallelStream().map(A -> {
487      final ELConceptDescription upperNeighbor = this.clone();
488      upperNeighbor.conceptNames.remove(A);
489      return upperNeighbor;
490    }), this.existentialRestrictions.entries().parallelStream().map(ER -> {
491      final ELConceptDescription upperNeighbor = this.clone();
492      upperNeighbor.existentialRestrictions.remove(ER.getKey(), ER.getValue());
493      ER
494          .getValue()
495          .upperNeighborsReduced()
496          .parallelStream()
497          .filter(
498              uER -> upperNeighbor.existentialRestrictions
499                  .entries()
500                  .parallelStream()
501                  .filter(otherER -> !otherER.equals(ER))
502                  .filter(otherER -> ER.getKey().equals(otherER.getKey()))
503                  .map(Entry::getValue)
504                  .noneMatch(uER::subsumes))
505          .map(uER -> Pair.of(ER.getKey(), uER))
506          .sequential()
507          .forEach(p -> upperNeighbor.existentialRestrictions.put(p.first(), p.second()));
508      return upperNeighbor;
509    })).filter(other::subsumes).findAny().orElse(null);
510  }

512  public final Set<ELConceptDescription> neighborhood(final int radius, final Signature sigma) {
513    if (radius < 0)
514      throw new IllegalArgumentException();
515    final Set<ELConceptDescription> next = Sets.newHashSet(this);
516    final Set<ELConceptDescription> neighborhood = Sets.newHashSet(this);
517    for (int k = 0; k < radius; k++) {
518      final Set<ELConceptDescription> news = Stream
519          .concat(
520              next.parallelStream().flatMap(C -> C.upperNeighborsReduced().parallelStream()),
521              next.parallelStream().flatMap(C -> C.lowerNeighborsReduced(sigma).parallelStream()))
522          .collect(Collectors.toSet());
523      next.clear();
524      next.addAll(news);
525      next.removeAll(neighborhood);
526      neighborhood.addAll(news);
527    }
528    return neighborhood;
529  }

577  public final Set<ELConceptDescription> lowerNeighbors(final Signature sigma) {
578    final ELConceptDescription C = ELConceptDescription.this.clone().reduce();
579    final Set<ELConceptDescription> lowerNeighbors = Sets.newConcurrentHashSet();
580    sigma.getConceptNames().parallelStream().filter(A -> !C.conceptNames.contains(A)).map(A -> {
581      final ELConceptDescription lowerNeighbor = C.clone();
582      lowerNeighbor.conceptNames.add(A);
583      return lowerNeighbor;
584    }).forEach(lowerNeighbors::add);
585    sigma.getRoleNames().parallelStream().forEach(r -> {
586      final Set<ELConceptDescription> filter = C.existentialRestrictions
587          .entries()
588          .parallelStream()
589          .filter(existentialRestriction -> existentialRestriction.getKey().equals(r))
590          .map(Entry::getValue)
591          .collect(Collectors.toSet());
592      if (filter.isEmpty()) {
593        final ELConceptDescription lowerNeighbor = C.clone();
594        lowerNeighbor.existentialRestrictions.put(r, ELConceptDescription.top());
595        lowerNeighbors.add(lowerNeighbor);
596      } else
597        recurseLowerNeighbors(sigma, r, C, Collections.singleton(ELConceptDescription.top()), filter, lowerNeighbors);
598    });
599    return lowerNeighbors;
600  }

602  private final void recurseLowerNeighbors(
603      final Signature sigma,
604      final IRI r,
605      final ELConceptDescription C,
606      final Set<ELConceptDescription> currentCandidates,
607      final Set<ELConceptDescription> filter,
608      final Set<ELConceptDescription> lowerNeighbors) {
609    final Set<ELConceptDescription> nextCandidates = Sets.newConcurrentHashSet();
610    currentCandidates.parallelStream().forEach(D -> {
611      if (!D.upperNeighborsReduced().parallelStream().allMatch(
612          U -> C.isSubsumedBy(ELConceptDescription.existentialRestriction(r, U))))
613        return;
614      else if (filter.parallelStream().anyMatch(F -> F.isSubsumedBy(D)))
615        D.lowerNeighbors(sigma).parallelStream().forEach(nextCandidates::add);
616      else {
617        final ELConceptDescription X = C.clone();
618        X.existentialRestrictions.put(r, D);
619        lowerNeighbors.add(X);
620      }
621    });
622    if (!nextCandidates.isEmpty())
623      recurseLowerNeighbors(sigma, r, C, nextCandidates, filter, lowerNeighbors);
624  }
