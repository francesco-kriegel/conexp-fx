150  public final boolean isInstanceOf(final OWLNamedIndividual i, final OWLClassExpression c) {
151    if (c.equals(df.getOWLThing()))
152      return true;
153    if (c.equals(df.getOWLNothing()))
154      return false;
155    if (c instanceof OWLClass)
156      return concepts.get(i, type, (OWLClass) c);
157    if (c instanceof OWLObjectSomeValuesFrom)
158      return roles.cut(i, (OWLObjectProperty) ((OWLObjectSomeValuesFrom) c).getProperty()).anyMatch(
159          j -> isInstanceOf(j, ((OWLObjectSomeValuesFrom) c).getFiller()));
160    if (c instanceof OWLObjectIntersectionOf)
161      return ((OWLObjectIntersectionOf) c).asConjunctSet().stream().allMatch(d -> isInstanceOf(i, d));
162    return false;
163  }

169  public final OWLClassExpression
170      getMMSC(final Set<OWLNamedIndividual> individuals, final int roleDepth, final DescriptionLogic dl) {
171    final Set<OWLClassExpression> conjuncts = new HashSet<OWLClassExpression>();
172    concepts.cut(individuals.stream(), type).forEach(new Consumer<OWLClass>() {
173
174      @Override
175      public void accept(OWLClass conceptName) {
176        conjuncts.add(conceptName);
177      }
178    });
179    if (roleDepth > 0)
180      for (OWLObjectProperty role : roles.getMs())
181        for (Constructor constructor : dl.constructors)
182          for (Set<OWLNamedIndividual> successor : minimalSuccessorSets(individuals, role, constructor))
183            switch (constructor) {
184            case EXISTENTIAL_RESTRICTION:
185              conjuncts.add(df.getOWLObjectSomeValuesFrom(role, getMMSC(successor, roleDepth - 1, dl)));
186              break;
187            case VALUE_RESTRICTION:
188              conjuncts.add(df.getOWLObjectAllValuesFrom(role, getMMSC(successor, roleDepth - 1, dl)));
189              break;
190            case QUALIFIED_AT_LEAST_RESTRICTION:
191//              conjuncts.add(df.getOWLObjectMinCardinality(cardinality, role, getMMSC(successor, roleDepth - 1, dl)));
192              break;
193            case UNQUALIFIED_AT_MOST_RESTRICTION:
194//              conjuncts.add(df.getOWLObjectMaxCardinality(cardinality, role, getMMSC(successor, roleDepth - 1, dl)));
195              break;
196            case CONJUNCTION:
197            default:
198            }
199    if (conjuncts.size() > 1)
200      return new OWLObjectIntersectionOfImpl(conjuncts);
201    if (conjuncts.size() == 1)
202      return conjuncts.iterator().next();
203    return df.getOWLThing();
204  }

206  public final OWLClassExpression
207      getMMSC_(final Set<String> individuals, final int roleDepth, final DescriptionLogic dl) {
208    return getMMSC(
209        individuals.stream().map(i -> df.getOWLNamedIndividual(IRI.create(i))).collect(Collectors.toSet()),
210        roleDepth,
211        dl);

214  private final Set<Set<OWLNamedIndividual>>
215      minimalSuccessorSets(Set<OWLNamedIndividual> individuals, OWLObjectProperty role, Constructor constructor) {
216    switch (constructor) {
217    case EXISTENTIAL_RESTRICTION:
218      final Set<OWLNamedIndividual> successors0 = new HashSet<OWLNamedIndividual>();
219      for (OWLNamedIndividual individual : individuals)
220        successors0.addAll(roles.cut(individual, role).collect(Collectors.toSet()));
221      // succerssors0 contains all r-successors of individuals in individuals
222//      final Set<Set<OWLNamedIndividual>> minSets =
223//          Sets
224//              .powerSet(successors0)
225//              .stream()
226//              .filter(
227//                  set -> set.stream().allMatch(
228//                      individual -> roles.row(role, individual).filter(individuals::contains).findFirst().isPresent()))
229//              .collect(Collectors.toSet());
230      final Set<Set<OWLNamedIndividual>> minSets = Sets.newHashSet(Sets.powerSet(successors0));
231      final Set<Set<OWLNamedIndividual>> t = new HashSet<Set<OWLNamedIndividual>>();
232      for (Set<OWLNamedIndividual> set : minSets) {
233        for (OWLNamedIndividual individual : individuals)
234          if (roles.cut(individual, role).noneMatch(set::contains))
235            t.add(set);
236      }
237      minSets.removeAll(t);
238      // minSets should here contain all subsets of successors0, that contain at least one r-successor
239      // for every individual in individuals
240//      final Set<Set<OWLNamedIndividual>> minSets = Sets.newHashSet(Sets.powerSet(successors0));
241      final Set<Set<OWLNamedIndividual>> nonMinSets = new HashSet<Set<OWLNamedIndividual>>();
242      for (Set<OWLNamedIndividual> x : minSets)
243        for (Set<OWLNamedIndividual> y : minSets)
244          if (!x.equals(y) && x.containsAll(y)) {
245            nonMinSets.add(x);
246            break;
247          }
248      minSets.removeAll(nonMinSets);
249      return minSets;
250    case VALUE_RESTRICTION:
251      final Set<OWLNamedIndividual> successors = new HashSet<OWLNamedIndividual>();
252      for (OWLNamedIndividual individual : individuals)
253        successors.addAll(roles.cut(individual, role).collect(Collectors.toSet()));
254      return Collections.singleton(successors);
255    case QUALIFIED_AT_LEAST_RESTRICTION:
256      return Collections.emptySet();
257    case UNQUALIFIED_AT_MOST_RESTRICTION:
258      return Collections.emptySet();
259    case CONJUNCTION:
260    default:
261      return Collections.emptySet();
262    }
263  }

265  public final MatrixContext<OWLNamedIndividual, OWLClassExpression>
266      toLogicalContext(final int roleDepth, final DescriptionLogic dl) {
267    MatrixContext<OWLNamedIndividual, OWLClassExpression> cxt =
268        new MatrixContext<OWLNamedIndividual, OWLClassExpression>(false);
269    cxt.rowHeads().addAll(concepts.getGs());
270    cxt.colHeads().addAll(createMMSCConjuncts(roleDepth, dl));
271    for (OWLNamedIndividual i : cxt.rowHeads())
272      for (OWLClassExpression c : cxt.colHeads())
273        if (isInstanceOf(i, c))
274          cxt.addFast(i, c);
275    return cxt;
276  }

278  private final List<OWLClassExpression> createMMSCConjuncts(final int roleDepth, final DescriptionLogic dl) {
279    final List<OWLClassExpression> mmscConjuncts = new ArrayList<OWLClassExpression>();
280//            (int) (1 + concepts.getWs().size() + roles.getMs().size()
281//            * Math.pow(2, concepts.getGs().size())));
282    mmscConjuncts.add(df.getOWLNothing());
283    mmscConjuncts.addAll(concepts.getWs());
284    if (roleDepth > 0) {
285      for (Set<OWLNamedIndividual> s : Sets.powerSet(concepts.getGs()))
286        if (!s.isEmpty()) {
287          final OWLClassExpression mmsc = getMMSC(s, roleDepth - 1, dl);
288          for (OWLObjectProperty role : roles.getMs()) {
289            for (Constructor constructor : dl.constructors)
290              switch (constructor) {
291              case EXISTENTIAL_RESTRICTION:
292                mmscConjuncts.add(df.getOWLObjectSomeValuesFrom(role, mmsc));
293                break;
294              case VALUE_RESTRICTION:
295                mmscConjuncts.add(df.getOWLObjectAllValuesFrom(role, mmsc));
296                break;
297              case QUALIFIED_AT_LEAST_RESTRICTION:
298              case UNQUALIFIED_AT_MOST_RESTRICTION:
299              case CONJUNCTION:
300              default:
301              }
302          }
303        }
304    }
305    return mmscConjuncts;
306  }

308  public static final Interpretation2 importFromRDFFile(final File rdfFile, final String baseURI)
309      throws RepositoryException, RDFParseException, IOException {
310    final Repository repo = new SailRepository(new MemoryStore());
311    final RepositoryConnection connection = repo.getConnection();
312    connection.add(rdfFile, baseURI, RDFFormat.forFileName(rdfFile.getName(), RDFFormat.RDFXML));
313    connection.commit();
314    connection.close();
315    return importFromRDFRepository(repo);
316  }

318  public static final Interpretation2 importFromRDFRepository(final Repository repo) throws RepositoryException {
319    final RepositoryConnection connection = repo.getConnection();
320
321    final Set<Resource> properties = new HashSet<Resource>();
322    final RepositoryResult<Statement> statements = connection.getStatements(null, RDF.TYPE, RDF.PROPERTY, true);
323    while (statements.hasNext())
324      properties.add(statements.next().getSubject());
325    statements.close();
326
327    final Set<Resource> classes = new HashSet<Resource>();
328    final RepositoryResult<Statement> statements2 = connection.getStatements(null, RDF.TYPE, RDFS.CLASS, true);
329    while (statements2.hasNext())
330      classes.add(statements2.next().getSubject());
331    statements2.close();
332
333    final Set<Resource> individuals = new HashSet<Resource>();
334    final RepositoryResult<Statement> statements3 = connection.getStatements(null, RDF.TYPE, null, true);
335    while (statements3.hasNext()) {
336      final Statement stmt = statements3.next();
337      if (!stmt.getObject().equals(RDF.PROPERTY) && !stmt.getObject().equals(RDFS.CLASS))
338        individuals.add(stmt.getSubject());
339    }
340    statements3.close();
341
342    final Interpretation2 i = new Interpretation2(individuals.size(), classes.size(), properties.size());
343    for (Resource individual : individuals)
344      i.addIndividual(individual.stringValue());
345    for (Resource clazz : classes)
346      i.addConcept(clazz.stringValue());
347    for (Resource property : properties)
348      i.addRole(property.stringValue());
349
350    final RepositoryResult<Statement> statements4 = connection.getStatements(null, RDF.TYPE, null, true);
351    while (statements4.hasNext()) {
352      final Statement stmt = statements4.next();
353      if (!stmt.getObject().equals(RDF.PROPERTY) && !stmt.getObject().equals(RDFS.CLASS))
354        i.addConceptAssertion(stmt.getSubject().stringValue(), stmt.getObject().stringValue());
355    }
356    statements4.close();
357
358    final RepositoryResult<Statement> statements5 = connection.getStatements(null, null, null, true);
359    while (statements5.hasNext()) {
360      final Statement stmt = statements5.next();
361      if (!stmt.getPredicate().equals(RDF.TYPE) && !stmt.getObject().equals(RDF.PROPERTY)
362          && !stmt.getObject().equals(RDFS.CLASS))
363        i.addRoleAssertion(
364            stmt.getSubject().stringValue(),
365            stmt.getPredicate().stringValue(),
366            stmt.getObject().stringValue());
367    }
368    statements5.close();
369
370    connection.close();
371    return i;
372  }
