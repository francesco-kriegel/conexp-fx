115  private static void baz2() {
116    final ELTBox tBox = new ELTBox();
117    tBox.getConceptInclusions().add(parseCI("A", "exists r.B"));
118    tBox.getConceptInclusions().add(parseCI("B", "C"));
119    tBox.getConceptInclusions().add(parseCI("exists r. C", "exists s. exists r.B"));
120    tBox.getConceptInclusions().add(parseCI("exists s. Top", "C"));
121    tBox.getConceptInclusions().add(parseCI("A and B", "exists r. C"));
122    final ELConceptDescription C = ELParser.read("A");
123    System.out.println(tBox);
124    System.out.println(C);
125    System.out.println(tBox.getCanonicalModel(C));
126    System.out.println(tBox.getCanonicalModelLutz(C));
127    for (int d = 0; d < 5; d++) {
128      final ELConceptDescription mss = tBox.getMostSpecificConsequence(C, d);
129      System.out.println(mss);
130      final ELConceptDescription mssL = tBox.getMostSpecificConsequenceLutz(C, d);
131      System.out.println(mssL);
132      System.out.println(mss.isEquivalentTo(mssL));
133    }
