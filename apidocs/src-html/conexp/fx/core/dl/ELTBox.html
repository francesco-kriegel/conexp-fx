040public class ELTBox {
041
042  private final Set<ELConceptInclusion> conceptInclusions;
043
044  public ELTBox() {
045    super();
046    this.conceptInclusions = new HashSet<ELConceptInclusion>();
047  }
048
049  public final Set<ELConceptInclusion> getConceptInclusions() {
050    return conceptInclusions;
051  }
052
053  private final class CanonicalModelBuilder {
054
055    private final ELConceptDescription                    C;
056    private final ELInterpretation2<ELConceptDescription> canmod;
057
058    private CanonicalModelBuilder(final ELConceptDescription C) {
059      super();
060      this.C = C;
061      this.canmod = new ELInterpretation2<>();
062    }
063
064    private final void insert(final ELConceptDescription D) {
065      insert(D, D);
066    }
067
068    private final void insert(final ELConceptDescription X, final ELConceptDescription D) {
069      for (IRI A : D.getConceptNames())
070        canmod.getConceptNameExtensionMatrix().add(X, A);
071      for (Entry<IRI, ELConceptDescription> rE : D.getExistentialRestrictions().entries()) {
072        canmod.getRoleNameExtensionMatrix(rE.getKey()).add(X, rE.getValue());
073        insert(rE.getValue());
074      }
075    }
076
077    private final ELInterpretation2<ELConceptDescription> buildAndGet() {
078      insert(C);
079      boolean changed = true;
080      while (changed) {
081        changed = false;
082        for (ELConceptInclusion ci : conceptInclusions)
083          for (ELConceptDescription o : new HashSet<>(canmod.getDomain())) {
084            final ELConceptDescription char1 =
085                canmod.getMostSpecificConceptDescription(Collections.singleton(o), ci.getSubsumee().roleDepth());
086            final ELConceptDescription char2 =
087                canmod.getMostSpecificConceptDescription(Collections.singleton(o), ci.getSubsumer().roleDepth());
088            if (char1.isSubsumedBy(ci.getSubsumee()) && !char2.isSubsumedBy(ci.getSubsumer())) {
089              insert(o, ci.getSubsumer());
090              changed = true;
091            }
092          }
093      }
094      return this.canmod;
095    }
096
097  }
098
099  public final ELInterpretation2<ELConceptDescription> getCanonicalModel(final ELConceptDescription C) {
100    return new CanonicalModelBuilder(C).buildAndGet();
101  }
102
103  public final ELConceptDescription getMostSpecificConsequence(final ELConceptDescription C, final int roleDepth) {
104    return getCanonicalModel(C).getMostSpecificConceptDescription(Collections.singleton(C), roleDepth);
105  }
106
107  private final class CanonicalModelBuilderLutz {
108
109    private final ELConceptDescription                    C;
110    private final ELInterpretation2<ELConceptDescription> canmod;
111    private final Set<Entry<IRI, ELConceptDescription>>   exsubT;
112    private final Set<ELConceptDescription>               domain;
113    private final Set<IRI>                                conceptNames = new HashSet<>();
114    private final Set<IRI>                                roleNames    = new HashSet<>();
115
116    private CanonicalModelBuilderLutz(final ELConceptDescription C) {
117      super();
118      this.C = C;
119      this.canmod = new ELInterpretation2<>();
120      this.exsubT = new HashSet<>();
121      this.domain = new HashSet<>();
122    }
123
124    private final void populate_exsubT(final ELConceptDescription X) {
125      exsubT.addAll(X.getExistentialRestrictions().entries());
126      for (ELConceptDescription Y : X.getExistentialRestrictions().values())
127        populate_exsubT(Y);
128    }
129
130    private final void populate_domain(final ELConceptDescription X) {
131      domain.addAll(X.getExistentialRestrictions().values());
132      for (ELConceptDescription Y : X.getExistentialRestrictions().values())
133        populate_domain(Y);
134    }
135
136    private final void populate_signature(final ELConceptDescription X) {
137      conceptNames.addAll(X.getConceptNames());
138      roleNames.addAll(X.getExistentialRestrictions().keys());
139      for (ELConceptDescription Y : X.getExistentialRestrictions().values())
140        populate_signature(Y);
141    }
142
143    private final ELInterpretation2<ELConceptDescription> buildAndGet() {
144      for (ELConceptInclusion ci : conceptInclusions) {
145        populate_exsubT(ci.getSubsumee());
146        populate_exsubT(ci.getSubsumer());
147      }
148      domain.add(C);
149      populate_domain(C);
150      for (Entry<IRI, ELConceptDescription> e : exsubT)
151        domain.add(e.getValue());
152      populate_signature(C);
153      for (ELConceptInclusion ci : conceptInclusions) {
154        populate_signature(ci.getSubsumee());
155        populate_signature(ci.getSubsumer());
156      }
157      for (ELConceptDescription D : domain) {
158        for (IRI A : conceptNames)
159          if (ELReasoner.isSubsumedBy(D, ELConceptDescription.conceptName(A), ELTBox.this))
160            canmod.getConceptNameExtensionMatrix().add(D, A);
161        for (ELConceptDescription E : domain)
162          for (IRI r : roleNames) {
163            final HashMap<IRI, ELConceptDescription> hashMap = new HashMap<>();
164            hashMap.put(r, E);
165            final Entry<IRI, ELConceptDescription> rE = hashMap.entrySet().iterator().next();
166            if ((exsubT.contains(rE)
167                && ELReasoner.isSubsumedBy(D, ELConceptDescription.existentialRestriction(rE), ELTBox.this))
168                || D.getExistentialRestrictions().containsEntry(r, E))
169              canmod.getRoleNameExtensionMatrix(r).add(D, E);
170          }
171      }
172      return canmod;
173    }
174
175  }
176
177  public final ELInterpretation2<ELConceptDescription> getCanonicalModelLutz(final ELConceptDescription C) {
178    return new CanonicalModelBuilderLutz(C).buildAndGet();
179  }
180
181  public final ELConceptDescription getMostSpecificConsequenceLutz(final ELConceptDescription C, final int roleDepth) {
182    return getCanonicalModelLutz(C).getMostSpecificConceptDescription(Collections.singleton(C), roleDepth);
183  }
184
185  @Override
186  public boolean equals(Object obj) {
187    if (obj == null)
188      return false;
189    if (!(obj instanceof ELTBox))
190      return false;
191    final ELTBox other = (ELTBox) obj;
192    return this.conceptInclusions.equals(other.conceptInclusions);
193  }
194
195  public final OWLOntology toOWLOntology() {
196    try {
197      final OWLOntologyManager om = OWLManager.createOWLOntologyManager();
198      final OWLDataFactory df = om.getOWLDataFactory();
199      final OWLOntology ontology = om.createOntology();
200      conceptInclusions
201          .parallelStream()
202          .forEach(
203              gci -> om
204                  .applyChange(
205                      new AddAxiom(
206                          ontology,
207                          df
208                              .getOWLSubClassOfAxiom(
209                                  gci.getSubsumee().clone().reduce().toOWLClassExpression(),
210                                  gci.getSubsumer().clone().reduce().toOWLClassExpression()))));
211      return ontology;
212    } catch (OWLOntologyCreationException e) {
213      throw new RuntimeException(e);
214    }
215  }
216
217  @Override
218  public int hashCode() {
219    return 23 * conceptInclusions.hashCode() + 99;
220  }
221
222  @Override
223  public String toString() {
224    return "EL-TBox " + conceptInclusions.toString();
225  }
