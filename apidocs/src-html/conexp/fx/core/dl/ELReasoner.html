039public class ELReasoner {
040
041  public static final boolean subsumes(final OWLClassExpression concept1, final OWLClassExpression concept2) {
042    return isSubsumedBy(concept2, concept1);
043  }
044
045  public static final boolean subsumes(final ELConceptDescription concept1, final ELConceptDescription concept2) {
046    return isSubsumedBy(concept2, concept1);
047  }
048
049  public static final boolean isSubsumedBy(final OWLClassExpression concept1, final OWLClassExpression concept2) {
050    return isSubsumedBy(ELConceptDescription.of(concept1), ELConceptDescription.of(concept2));
051  }
052
053  /**
054   * @param concept1
055   * @param concept2
056   * @return true, iff concept1 is subsumed by concept2 (w.r.t. empty TBox)
057   */
058  public static final boolean isSubsumedBy(final ELConceptDescription concept1, final ELConceptDescription concept2) {
059    if (concept1.isBot())
060      return true;
061    if (concept2.isTop())
062      return true;
063    if (!concept1.getConceptNames().containsAll(concept2.getConceptNames()))
064      return false;
065    return concept2.getExistentialRestrictions().entries().parallelStream().allMatch(
066        existentialRestriction2 -> concept1.getExistentialRestrictions().entries().parallelStream().anyMatch(
067            existentialRestriction1 -> existentialRestriction2.getKey().equals(existentialRestriction1.getKey())
068                && isSubsumedBy(existentialRestriction1.getValue(), existentialRestriction2.getValue())));
069  }
070
071  public static final boolean
072      isSubsumedBy(final ELConceptDescription concept1, final ELConceptDescription concept2, final ELTBox tBox) {
073    return isSubsumedBy(concept1.toOWLClassExpression(), concept2.toOWLClassExpression(), tBox.toOWLOntology());
074  }
075
076  private static int dummy = 1618;
077
078  private static final int nextDummy() {
079    return dummy++;
080  }
081
082  public static final boolean
083      isSubsumedBy(final OWLClassExpression concept1, final OWLClassExpression concept2, final OWLOntology ontology) {
084    final OWLOntologyManager om = OWLManager.createOWLOntologyManager();
085    final OWLDataFactory df = OWLManager.getOWLDataFactory();
086
087    final OWLClass c1 = df.getOWLClass(IRI.create("dummy" + nextDummy()));
088    final OWLClass c2 = df.getOWLClass(IRI.create("dummy" + nextDummy()));
089    final OWLSubClassOfAxiom ax1 = df.getOWLSubClassOfAxiom(concept1, c1);
090    final OWLSubClassOfAxiom _ax1 = df.getOWLSubClassOfAxiom(c1, concept1);
091    final OWLSubClassOfAxiom ax2 = df.getOWLSubClassOfAxiom(c2, concept2);
092    final OWLSubClassOfAxiom _ax2 = df.getOWLSubClassOfAxiom(concept2, c2);
093    om.applyChange(new AddAxiom(ontology, ax1));
094    om.applyChange(new AddAxiom(ontology, ax2));
095    om.applyChange(new AddAxiom(ontology, _ax1));
096    om.applyChange(new AddAxiom(ontology, _ax2));
097
098    final OWLReasoner elk = new ElkReasonerFactory().createReasoner(ontology);
099    elk.flush();
100    elk.precomputeInferences(InferenceType.CLASS_HIERARCHY);
101
102    final boolean result =
103        elk.getEquivalentClasses(c2).contains(c1) || elk.getSubClasses(c2, false).getFlattened().contains(c1);
104    elk.dispose();
105    om.applyChange(new RemoveAxiom(ontology, ax1));
106    om.applyChange(new RemoveAxiom(ontology, ax2));
107    om.applyChange(new RemoveAxiom(ontology, _ax1));
108    om.applyChange(new RemoveAxiom(ontology, _ax2));
109    return result;
110//    return elk.isEntailed(df.getOWLSubClassOfAxiom(concept1, concept2));
111  }
112
113  public static final boolean isSubsumedBy(
114      final ELConceptDescription concept1,
115      final ELConceptDescription concept2,
116      final OWLOntology ontology) {
117    return isSubsumedBy(concept1.toOWLClassExpression(), concept2.toOWLClassExpression(), ontology);
118  }
