117  public static final ELConceptDescription lcs(final ELConceptDescription C, final ELConceptDescription D) {
118    return lcs(Sets.newHashSet(C, D));
119  }
120
121  public static final ELConceptDescription lcs(final Set<ELConceptDescription> Cs) {
122    Cs.parallelStream().forEach(ELConceptDescription::reduce);
123    final Set<ELConceptDescription> Ds = Collections3.representatives(Cs, (X, Y) -> X.isEquivalentTo(Y));
124    if (Ds.isEmpty())
125      return ELConceptDescription.bot();
126    else if (Ds.size() == 1)
127      return Ds.iterator().next().clone();
128    else {
129      final ELConceptDescription lcs = new ELConceptDescription();
130      final Iterator<ELConceptDescription> it = Ds.iterator();
131      final ELConceptDescription D = it.next();
132      it.remove();
133      final Set<IRI> commonConceptNames = D
134          .getConceptNames()
135          .parallelStream()
136          .filter(A -> Ds.parallelStream().map(ELConceptDescription::getConceptNames).allMatch(As -> As.contains(A)))
137          .collect(Collectors.toSet());
138      lcs.getConceptNames().addAll(commonConceptNames);
139      final Set<IRI> commonRoleNames = D
140          .getExistentialRestrictions()
141          .keySet()
142          .parallelStream()
143          .filter(
144              r -> Ds.parallelStream().map(ELConceptDescription::getExistentialRestrictions).allMatch(
145                  ERs -> ERs.keySet().parallelStream().anyMatch(r::equals)))
146          .collect(Collectors.toSet());
147      Ds.add(D);
148      commonRoleNames
149          .parallelStream()
150          .map(
151              r -> Pair.of(
152                  r,
153                  Sets
154                      .cartesianProduct(
155                          Ds
156                              .parallelStream()
157                              .map(ELConceptDescription::getExistentialRestrictions)
158                              .map(m -> m.get(r))
159                              .map(HashSet::new)
160                              .collect(Collectors.toList()))
161                      .parallelStream()
162                      .map(HashSet::new)
163                      .map(ELLeastCommonSubsumer::lcs)
164                      .map(ELConceptDescription::reduce)
165                      .collect(Collectors.toSet())))
166          .sequential()
167          .forEach(p -> lcs.getExistentialRestrictions().putAll(p.x(), p.y()));;
168      return lcs.clone().reduce();
169    }
170  }

172  public static final OWLClassExpression of(final OWLClassExpression concept1, final OWLClassExpression concept2) {
173    return ELLeastCommonSubsumer
174        .lcs(ELConceptDescription.of(concept1), ELConceptDescription.of(concept2))
175        .toOWLClassExpression();
176  }
177
178  public static final OWLClassExpression of(final OWLClassExpression... concepts) {
179    if (concepts.length == 0)
180      return df.getOWLThing();
181    if (concepts.length == 1)
182      return concepts[0];
183    if (concepts.length == 2)
184      return of(concepts[0], concepts[1]);
185    return of(Arrays.asList(concepts));
186  }
187
188  public static final OWLClassExpression of(final Collection<OWLClassExpression> concepts) {
189    if (concepts.isEmpty())
190      return df.getOWLThing();
191    final Iterator<OWLClassExpression> it = concepts.iterator();
192    if (concepts.size() == 1)
193      return it.next();
194    if (concepts.size() == 2)
195      return of(it.next(), it.next());
196    OWLClassExpression lcs = it.next();
197    while (it.hasNext())
198      lcs = of(lcs, it.next());
199    return lcs;
200  }
