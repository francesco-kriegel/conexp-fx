043public class ELInterpretation2<I> {
044
045  private final MatrixRelation<I, IRI>         conceptNameExtensionMatrix;
046  private final Map<IRI, MatrixRelation<I, I>> roleNameExtensionMatrix;
047
048  public ELInterpretation2() {
049    super();
050    this.conceptNameExtensionMatrix = new MatrixRelation<>(false);
051    this.roleNameExtensionMatrix = new HashMap<>();
052  }
053
054  public final MatrixRelation<I, IRI> getConceptNameExtensionMatrix() {
055    return this.conceptNameExtensionMatrix;
056  }
057
058  public final MatrixRelation<I, I> getRoleNameExtensionMatrix(final IRI roleName) {
059    return this.roleNameExtensionMatrix.computeIfAbsent(roleName, __ -> new MatrixRelation<>(true));
060  }
061
062  public final Set<I> getDomain() {
063    return roleNameExtensionMatrix
064        .values()
065        .stream()
066        .reduce((Set<I>) conceptNameExtensionMatrix.rowHeads(), (s, m) -> Sets.union(s, m.colHeads()), Sets::union);
067  }
068
069  private final Predicate<I> satisfiesAllExistentialRestrictions(final ELConceptDescription conceptDescription) {
070    return i -> conceptDescription
071        .getExistentialRestrictions()
072        .entries()
073        .parallelStream()
074        .allMatch(
075            e -> roleNameExtensionMatrix
076                .get(e.getKey())
077                .row(i)
078                .parallelStream()
079                .anyMatch(j -> isInExtensionOf(j, e.getValue())));
080  }
081
082  public final Set<I> getExtension(final ELConceptDescription conceptDescription) {
083    if (conceptDescription.isBot())
084      return Collections.emptySet();
085    else if (conceptDescription.isTop())
086      return new HashSet<>(this.getDomain());
087    else
088      return this.conceptNameExtensionMatrix
089          .colAnd(conceptDescription.getConceptNames())
090          .parallelStream()
091          .filter(satisfiesAllExistentialRestrictions(conceptDescription))
092          .collect(Collectors.toSet());
093  }
094
095  public final boolean isInExtensionOf(final I i, final ELConceptDescription conceptDescription) {
096    if (conceptDescription.isBot())
097      return false;
098    else if (conceptDescription.isTop())
099      return true;
100    else
101      return this.conceptNameExtensionMatrix.colAnd(conceptDescription.getConceptNames()).contains(i)
102          && satisfiesAllExistentialRestrictions(conceptDescription).test(i);
103  }
104
105  public final boolean models(final ELConceptInclusion conceptInclusion) {
106    return getDomain()
107        .parallelStream()
108        .allMatch(
109            Predicates
110                .<I> not(i -> isInExtensionOf(i, conceptInclusion.getSubsumee()))
111                .or(i -> isInExtensionOf(i, conceptInclusion.getSubsumer())));
112  }
113
114  public final boolean models(final ELTBox tBox) {
115    return tBox.getConceptInclusions().parallelStream().allMatch(this::models);
116  }
117
118  public final ELConceptDescription getMostSpecificConceptDescription(final Set<I> objects, final int roleDepth) {
119    if (roleDepth < 0)
120      throw new IllegalArgumentException();
121    else if (objects.isEmpty())
122      return ELConceptDescription.bot();
123    else {
124      final ELConceptDescription mmsc = new ELConceptDescription();
125      mmsc.getConceptNames().addAll(this.conceptNameExtensionMatrix.rowAnd(objects));
126      if (roleDepth > 0) {
127        for (Entry<IRI, MatrixRelation<I, I>> e : roleNameExtensionMatrix.entrySet()) {
128          final IRI roleName = e.getKey();
129          final Set<Set<I>> hypergraph = objects
130              .parallelStream()
131              .map(o -> e.getValue().rowHeads().contains(o) ? e.getValue().row(o) : Collections.<I> emptySet())
132              .collect(Collectors.toSet());
133          for (Set<I> mhs : getMinimalHittingSets(hypergraph)) {
134            mmsc.getExistentialRestrictions().put(roleName, getMostSpecificConceptDescription(mhs, roleDepth - 1));
135          }
136        }
137      }
138      return mmsc.reduce();
139    }
140  }
141
142  public static final <T> Set<Set<T>> getMinimalHittingSets(final Set<Set<T>> hypergraph) {
143    if (hypergraph.isEmpty())
144      return Collections.<Set<T>> emptySet();
145    else {
146      final Set<Set<T>> identity = Collections.singleton(Collections.emptySet());
147      final BiFunction<Set<Set<T>>, Set<T>, Set<Set<T>>> accumulator = (hittingSets, factor) -> hittingSets
148          .parallelStream()
149          .flatMap(hittingSet -> factor.parallelStream().map(element -> {
150            final Set<T> newHittingSet = new HashSet<>(hittingSet);
151            newHittingSet.add(element);
152            return newHittingSet;
153          }))
154          .collect(Collectors.toSet());
155      final BinaryOperator<Set<Set<T>>> combiner = (hittingSets1, hittingSets2) -> hittingSets1
156          .parallelStream()
157          .flatMap(hittingSet1 -> hittingSets2.parallelStream().map(hittingSet2 -> {
158            final Set<T> newHittingSet = new HashSet<>(hittingSet1);
159            newHittingSet.addAll(hittingSet2);
160            return newHittingSet;
161          }))
162          .collect(Collectors.toSet());
163      final Set<Set<T>> hittingSets = hypergraph.parallelStream().reduce(identity, accumulator, combiner);
164      final Set<Set<T>> nonMinimalHittingSets = Sets.newConcurrentHashSet();
165      hittingSets.parallelStream().forEach(hittingSet1 -> {
166        if (hittingSets
167            .parallelStream()
168            .anyMatch(hittingSet2 -> hittingSet1.containsAll(hittingSet2) && !hittingSet2.containsAll(hittingSet1)))
169          nonMinimalHittingSets.add(hittingSet1);
170      });
171      hittingSets.removeAll(nonMinimalHittingSets);
172      return hittingSets;
173    }
174  }
175
176  @Override
177  public String toString() {
178    return "Interpretation\r\n" + conceptNameExtensionMatrix + "\r\n" + roleNameExtensionMatrix;
179  }
