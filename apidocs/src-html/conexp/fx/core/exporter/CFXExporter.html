026import java.io.BufferedWriter;
027import java.io.File;
028import java.io.FileWriter;
029import java.io.IOException;
030import java.util.Map;
031import java.util.Map.Entry;
032
033import javafx.geometry.Point3D;
034
035import org.jsoup.nodes.Document;
036import org.jsoup.nodes.Element;
037
038import conexp.fx.core.context.MatrixContext;
039import conexp.fx.core.layout.AdditiveConceptLayout;
040
041public class CFXExporter<G, M> {
042
043  public static <G, M> void export(
044      MatrixContext<G, M> context,
045      Map<Integer, Integer> domainPermutation,
046      Map<Integer, Integer> codomainPermutation,
047      AdditiveConceptLayout<G, M> layout,
048      File file) {
049    Document xml = new Document("");
050    final Element domainEl = xml.appendElement("domain");
051//    domainEl.attr("type", formalContext.getDomainType() == null ? "unknown" : formalContext.getDomainType().toString());
052    domainEl.attr("size", String.valueOf(context.rowHeads().size()));
053    for (int domainIndex = 0; domainIndex < context.rowHeads().size(); domainIndex++) {
054      final int permIndex =
055          domainPermutation == null || !domainPermutation.containsKey(domainIndex) ? domainIndex : domainPermutation
056              .get(domainIndex);
057      final G object = context.rowHeads().get(permIndex);
058      final Element objectEl = domainEl.appendElement("object");
059      objectEl.attr("index", String.valueOf(permIndex));
060      objectEl.attr("name", object.toString());
061      objectEl.attr("selected", "true");// String.valueOf(formalContext.getSelectedObjects().contains(object)));
062    }
063    final Element codomainEl = xml.appendElement("codomain");
064//    codomainEl.attr("type", formalContext.getCodomainType() == null ? "unknown" : formalContext
065//        .getCodomainType()
066//        .toString());
067    codomainEl.attr("size", String.valueOf(context.colHeads().size()));
068    for (int codomainIndex = 0; codomainIndex < context.colHeads().size(); codomainIndex++) {
069      int permIndex =
070          codomainPermutation == null || !codomainPermutation.containsKey(codomainIndex) ? codomainIndex
071              : codomainPermutation.get(codomainIndex);
072      final M attribute = context.colHeads().get(permIndex);
073      final Element attributeEl = codomainEl.appendElement("attribute");
074      attributeEl.attr("index", String.valueOf(permIndex));
075      attributeEl.attr("name", attribute.toString());
076      attributeEl.attr("selected", String.valueOf(context.selectedAttributes().contains(attribute)));
077    }
078    final Element contextEl = xml.appendElement("context");
079    for (int domainIndex = 0; domainIndex < context.rowHeads().size(); domainIndex++) {
080      final int dpermIndex =
081          domainPermutation == null || !domainPermutation.containsKey(domainIndex) ? domainIndex : domainPermutation
082              .get(domainIndex);
083      final G object = context.rowHeads().get(dpermIndex);
084      for (int codomainIndex = 0; codomainIndex < context.colHeads().size(); codomainIndex++) {
085        int cpermIndex =
086            codomainPermutation == null || !codomainPermutation.containsKey(codomainIndex) ? codomainIndex
087                : codomainPermutation.get(codomainIndex);
088        final M attribute = context.colHeads().get(cpermIndex);
089        if (context._contains(dpermIndex, cpermIndex)) {
090          final Element incidenceEl = contextEl.appendElement("incidence");
091          incidenceEl.attr("object", object.toString());
092          incidenceEl.attr("attribute", attribute.toString());
093        } else {
094          final boolean upArrow = context.UpArrows.contains(object, attribute);
095          final boolean downArrow = context.DownArrows.contains(object, attribute);
096          if (upArrow && downArrow) {
097            final Element arrowsEl = contextEl.appendElement("both-arrows");
098            arrowsEl.attr("object", object.toString());
099            arrowsEl.attr("attribute", attribute.toString());
100          } else if (upArrow) {
101            final Element arrowsEl = contextEl.appendElement("up-arrow");
102            arrowsEl.attr("object", object.toString());
103            arrowsEl.attr("attribute", attribute.toString());
104          } else if (downArrow) {
105            final Element arrowsEl = contextEl.appendElement("down-arrow");
106            arrowsEl.attr("object", object.toString());
107            arrowsEl.attr("attribute", attribute.toString());
108          }
109        }
110      }
111    }
112    final Element latticeEl = xml.appendElement("lattice");
113    for (Entry<M, Point3D> attributeSeed : layout.seedsM.entrySet()) {
114      final Element seedEl = latticeEl.appendElement("attribute-seed");
115      seedEl.attr("attribute", attributeSeed.getKey().toString());
116      seedEl.attr("x", String.valueOf(attributeSeed.getValue().getX()));
117      seedEl.attr("y", String.valueOf(attributeSeed.getValue().getY()));
118      seedEl.attr("z", String.valueOf(attributeSeed.getValue().getZ()));
119    }
120    try {
121      if (!file.exists()) {
122        if (!file.getParentFile().exists())
123          file.mkdirs();
124        file.createNewFile();
125      }
126      final FileWriter fileWriter = new FileWriter(file);
127      final BufferedWriter bufferedWriter = new BufferedWriter(fileWriter);
128      bufferedWriter.write(xml.toString());
129      bufferedWriter.close();
130      fileWriter.close();
131    } catch (IOException e) {
132      // TODO Auto-generated catch block
133      System.err.println("Unable to write CFX-File to " + file.toString());
134      e.printStackTrace();
135    }
136  }
137}
