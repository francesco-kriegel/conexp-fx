026import java.io.File;

036import conexp.fx.core.context.MatrixContext;
037
038public class CEXImporter {
039
040  public static void importt(MatrixContext<String, String> _context, Map<String, Point3D> seedMap, File file) {
041    try {
042      final Element xml = Jsoup.parse(file, null).body();
043      final Element context =
044          xml
045              .getElementsByTag("ConceptualSystem")
046              .first()
047              .getElementsByTag("Contexts")
048              .first()
049              .getElementsByTag("Context")
050              .first();
051      final Element attributes = context.getElementsByTag("Attributes").first();
052      final Element objects = context.getElementsByTag("Objects").first();
053      final Map<Integer, String> attIdMap = new HashMap<Integer, String>();
054      for (Element attributeEl : attributes.getElementsByTag("Attribute")) {
055        final Integer id = Integer.valueOf(attributeEl.attr("Identifier"));
056        final String attribute = attributeEl.getElementsByTag("Name").first().text();
057        attIdMap.put(id, attribute);
058        _context.colHeads().add(attribute);
059      }
060      int i = 0;
061      for (Element objectEl : objects.getElementsByTag("Object")) {
062        final String object = objectEl.getElementsByTag("Name").first().text();
063        _context.rowHeads().add(object);
064        System.out.println(i++ + "reading object " + object);
065        for (Element hasAttributeEl : objectEl.getElementsByTag("Intent").first().getElementsByTag("HasAttribute")) {
066          _context.addFastSilent(object, attIdMap.get(Integer.valueOf(hasAttributeEl.attr("AttributeIdentifier"))));
067        }
068      }
069//          if (seedMap != null) {
070//            // TODO extract seed vectors
071//              for () {
072//                  final String attribute = ;
073//                  final Double x = ;
074//                  final Double y = ;
075//                  final Double z = ;
076//                  final Point3D point3d = new Point3D(x, y, z);
077//                  seedMap.put(attribute, point3d);
078//              }
079//          }
080      _context.pushAllChangedEvent();
081    } catch (IOException e) {
082      // TODO Auto-generated catch block
083      System.err.println("Unable to parse Concept Explorer file from " + file.toString());
084      e.printStackTrace();
085    }
086  }
087}
