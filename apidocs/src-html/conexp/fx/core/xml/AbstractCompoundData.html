026import java.util.Collection;
027import java.util.Collections;
028import java.util.LinkedList;
029import java.util.List;
030import java.util.Map;
031import java.util.Set;
032import java.util.concurrent.ConcurrentHashMap;
033
034import org.jsoup.nodes.Element;
035
036import com.google.common.base.Function;
037import com.google.common.collect.Iterables;
038
039import conexp.fx.core.collections.Pair;
040
041public abstract class AbstractCompoundData extends Data<Map<String, Data<?>>> implements Map<String, Data<?>> {
042
043  private static final Function<Element, String> ELEMENT_TO_TEXT_FUNCTION = new Function<Element, String>() {
044
045                                                                            public final String apply(
046                                                                                final Element element) {
047                                                                              return element.text();
048                                                                            }
049                                                                          };
050
051  public AbstractCompoundData(final Datatype type, final String key) {
052    super(type, key, new ConcurrentHashMap<String, Data<?>>());
053  }
054
055  public AbstractCompoundData(final Datatype type, final String key, final Map<String, Data<?>> value) {
056    super(type, key, new ConcurrentHashMap<String, Data<?>>(value));
057  }
058
059  public AbstractCompoundData(final Datatype type, final String key, final Element element, final Metadata metadata)
060      throws NullPointerException, IndexOutOfBoundsException {
061    super(type, key, readDataFromElement(element, metadata));
062  }
063
064  protected synchronized static final Map<String, Data<?>> readDataFromElement(
065      final Element element,
066      final Metadata metadata) {
067    final Map<String, Data<?>> map = new ConcurrentHashMap<String, Data<?>>();
068    for (Entry<String, Pair<Datatype, Metadata>> entry : metadata.getKeyDatatypeMap().entrySet()) {
069      final String key = entry.getKey();
070      final Datatype type = entry.getValue().first();
071      final Metadata _metadata = entry.getValue().second();
072      final Element data = JsoupUtil.firstElement(element, Key.toArray(key));
073      switch (type) {
074      case BOOLEAN:
075        map.put(key, new BooleanData(key, data.text()));
076        break;
077      case INTEGER:
078        map.put(key, new IntegerData(key, data.text()));
079        break;
080      case FLOAT:
081        map.put(key, new FloatData(key, data.text()));
082        break;
083      case STRING:
084        try {
085          map.put(key, new StringData(key, data.text()));
086        } catch (Exception e) {
087          System.err.println(data);
088          e.printStackTrace();
089        }
090        break;
091      case COMPOUND:
092        map.put(key, new CompoundData(key, data, entry.getValue().second()));
093        break;
094      case BOOLEAN_LIST:
095        map.put(key, new BooleanListData(key, _metadata.getSubkey(), null, readStringDataFromElement(data, _metadata)));
096        break;
097      case INTEGER_LIST:
098        map.put(key, new IntegerListData(key, _metadata.getSubkey(), null, readStringDataFromElement(data, _metadata)));
099        break;
100      case FLOAT_LIST:
101        map.put(key, new FloatListData(key, _metadata.getSubkey(), null, readStringDataFromElement(data, _metadata)));
102        break;
103      case STRING_LIST:
104        map.put(key, new StringListData(key, _metadata.getSubkey(), readStringDataFromElement(data, _metadata)));
105        break;
106      case COMPOUND_LIST:
107        map.put(key, new CompoundListData(key, _metadata.getSubkey(), data, _metadata));
108      case METADATA:
109      default:
110      }
111    }
112    return map;
113  }
114
115  protected static List<String> readStringDataFromElement(final Element element, final Metadata metadata) {
116    List<String> list = new LinkedList<String>();
117    for (String value : Iterables.transform(
118        JsoupUtil.childrenByTag(element, metadata.getSubkey()),
119        ELEMENT_TO_TEXT_FUNCTION))
120      list.add(value);
121    return Collections.synchronizedList(list);
122  }
123
124  @Override
125  public final boolean isEmpty() {
126    return value.isEmpty();
127  }
128
129  @Override
130  public final int size() {
131    return value.size();
132  }
133
134  @Override
135  public final void clear() {
136    value.clear();
137  }
138
139  @Override
140  public final boolean containsKey(Object key) {
141    return value.containsKey(key);
142  }
143
144  @Override
145  public final boolean containsValue(Object value) {
146    return this.value.containsValue(value);
147  }
148
149  @Override
150  public final Data<?> get(Object key) {
151    return value.get(key);
152  }
153
154  @Override
155  public final Data<?> put(String key, Data<?> value) {
156    return this.value.put(key, value);
157  }
158
159  @Override
160  public final void putAll(Map<? extends String, ? extends Data<?>> m) {
161    value.putAll(m);
162  }
163
164  @Override
165  public final Data<?> remove(Object key) {
166    return value.remove(key);
167  }
168
169  @Override
170  public final Set<String> keySet() {
171    return value.keySet();
172  }
173
174  @Override
175  public final Collection<Data<?>> values() {
176    return value.values();
177  }
178
179  @Override
180  public final Set<Entry<String, Data<?>>> entrySet() {
181    return value.entrySet();
182  }
