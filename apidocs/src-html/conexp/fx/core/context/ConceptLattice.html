047public final class ConceptLattice<G, M> extends MatrixRelation<Concept<G, M>, Concept<G, M>> {

049  public final MatrixContext<G, M>                     context;
050  public final ObservableMap<G, Concept<G, M>>         objectConcepts    =
051                                                                             FXCollections
052                                                                                 .observableMap(new ConcurrentHashMap<G, Concept<G, M>>());
053  public final ObservableMap<M, Concept<G, M>>         attributeConcepts =
054                                                                             FXCollections
055                                                                                 .observableMap(new ConcurrentHashMap<M, Concept<G, M>>());
056  private MatrixRelation<Concept<G, M>, Concept<G, M>> order;

058  public ConceptLattice(final MatrixContext<G, M> context) {
059    super(true);
060    this.context = context;
061    order = order();
062    addEventHandler(new RelationEventHandler<Concept<G, M>, Concept<G, M>>() {
063
064      @Override
065      public final void handle(final RelationEvent<Concept<G, M>, Concept<G, M>> event) {
066        order = order();
067      }
068    }, RelationEvent.ALL_CHANGED);
069  }

071  public final Set<Concept<G, M>> lowerNeighbors(final Concept<G, M> concept) {
072    return col(concept);
073  }
074
075  public final Set<Concept<G, M>> upperNeighbors(final Concept<G, M> concept) {
076    return row(concept);
077  }
078
079  // lower
080  public final Set<Concept<G, M>> ideal(final Concept<G, M> concept) {
081    return order.col(concept);
082  }
083
084  // upper
085  public final Set<Concept<G, M>> filter(final Concept<G, M> concept) {
086    return order.row(concept);
087  }
088
089  public final Set<Concept<G, M>> interval(final Concept<G, M> lower, final Concept<G, M> upper) {
090    return Sets.intersection(filter(lower), ideal(upper));
091  }
092
093  public final Set<Concept<G, M>> complement(final Set<Concept<G, M>> concepts) {
094    return Sets.difference(rowHeads(), concepts);
095  }
096
097  @SafeVarargs
098  public final Concept<G, M> supremum(final Concept<G, M>... concepts) {
099    final Set<M> intent = new HashSet<M>(context.colHeads());
100    for (Concept<G, M> concept : concepts)
101      intent.retainAll(concept.intent());
102    final Set<G> extent = new HashSet<G>(context.colAnd(intent));
103    return new Concept<G, M>(extent, intent);
104  }
105
106  public final Concept<G, M> supremum(final Iterable<Concept<G, M>> concepts) {

112  }

139  /**
140   * The method computes the luxenburger base for partial implications, whose confidence is at least minConfidence. If
141   * sorted is set to true, then the partial implications are sorted according to descending confidence, and a List is
142   * returned. Otherwise an unsorted Set is returned.
143   * 
144   * @param minConfidence
145   * @param sorted
146   * @return a collection of implications
147   */
148  public final Collection<Implication<G, M>> luxenburgerBase(final double minConfidence, final boolean sorted) {
149    if (minConfidence < 0d || minConfidence > 1d)
150      throw new IllegalArgumentException("Confidence must be in range [0,1].");
151    Stream<Implication<G, M>> s = new LuxenburgerBaseExtractor().get().parallelStream();
152    if (minConfidence > 0d)
153      s = s.filter(i -> i.getConfidence() >= minConfidence);
154    if (sorted)
155      return s.sorted((i, j) -> (int) Math.signum(j.getConfidence() - i.getConfidence())).collect(Collectors.toList());
156    return s.collect(Collectors.toSet());
157  }

178  private final class LuxenburgerBaseExtractor {
179
180    private final Set<Implication<G, M>> set = Collections
181                                                 .newSetFromMap(new ConcurrentHashMap<Implication<G, M>, Boolean>());
182
183    private final Set<Implication<G, M>> get() {
184      recursion(context.selection.topConcept());
185      return set;
186    }
187
188    private final void recursion(final Concept<G, M> concept) {
189      col(concept).parallelStream().forEach(
190          lowerNeighbor -> {
191            set.add(new Implication<G, M>(concept.getIntent(), Sets.newHashSet(Sets.difference(
192                lowerNeighbor.getIntent(),
193                concept.getIntent())), lowerNeighbor.getExtent(), ((double) lowerNeighbor.getExtent().size())
194                / ((double) concept.getExtent().size())));
195            recursion(lowerNeighbor);
196          });
197    }
