025import java.util.Collection;
026import java.util.Iterator;
027import java.util.Set;
028
029import com.google.common.base.Function;
030import com.google.common.collect.Iterators;
031
032import de.tudresden.inf.tcs.fcaapi.Expert;
033import de.tudresden.inf.tcs.fcaapi.FCAImplication;
034import de.tudresden.inf.tcs.fcaapi.FCAObject;
035import de.tudresden.inf.tcs.fcaapi.exception.IllegalAttributeException;
036import de.tudresden.inf.tcs.fcaapi.exception.IllegalObjectException;
037import de.tudresden.inf.tcs.fcaapi.utils.IndexedSet;
038import de.tudresden.inf.tcs.fcalib.AbstractContext;
039import de.tudresden.inf.tcs.fcalib.FullObject;
040import de.tudresden.inf.tcs.fcalib.FullObjectDescription;
041
042public final class FCALibContext<G, M> extends AbstractContext<M, G, FCAObject<M, G>> {
043
044  private final MatrixContext<G, M>     cxt;
045  private Expert<M, G, FCAObject<M, G>> expert;
046
047  public FCALibContext(final MatrixContext<G, M> cxt) {
048    super();
049    this.cxt = cxt;
050    this.getAttributes().addAll(cxt.colHeads());
051  }
052
053  @Override
054  public FCAObject<M, G> getObject(G id) {
055    return new FullObject<M, G>(id, cxt.row(id));
056  }
057
058  @Override
059  public FCAObject<M, G> getObjectAtIndex(int index) {
060    return getObject(cxt.rowHeads().get(index));
061  }
062
063  @Override
064  public boolean objectHasAttribute(FCAObject<M, G> object, M attribute) {
065    return cxt.contains(object.getIdentifier(), attribute);
066  }
067
068  @Override
069  public Set<FCAImplication<M>> getDuquenneGuiguesBase() {
070    return null;
071  }
072
073  @Override
074  public Expert<M, G, FCAObject<M, G>> getExpert() {
075    return expert;
076  }
077
078  @Override
079  public void setExpert(Expert<M, G, FCAObject<M, G>> e) {
080    expert = e;
081  }
082
083  @Override
084  public IndexedSet<FCAObject<M, G>> getObjects() {
085    return new IndexedSet<FCAObject<M, G>>() {
086
087      @Override
088      public int size() {
089        return cxt.rowHeads().size();
090      }
091
092      @Override
093      public boolean isEmpty() {
094        return cxt.rowHeads().isEmpty();
095      }
096
097      @Override
098      public boolean contains(Object o) {
099        if (o instanceof FCAObject)
100          return cxt.rowHeads().contains(((FCAObject<?, ?>) o).getIdentifier());
101        return cxt.rowHeads().contains(o);
102      }
103
104      @Override
105      public Iterator<FCAObject<M, G>> iterator() {
106        return Iterators.transform(cxt.rowHeads().iterator(), new Function<G, FCAObject<M, G>>() {
107
108          @Override
109          public FCAObject<M, G> apply(G object) {
110            return getObject(object);
111          }
112        });
113      }
114
115      @Override
116      public Object[] toArray() {
117        return cxt.rowHeads().toArray();
118      }
119
120      @Override
121      public <T> T[] toArray(T[] a) {
122        return cxt.rowHeads().toArray(a);
123      }
124
125      @Override
126      public boolean add(FCAObject<M, G> e) {
127        return cxt.rowHeads().add(e.getIdentifier())
128            && cxt.row(e.getIdentifier()).addAll(((FullObjectDescription<M>) e.getDescription()).getAttributes());
129      }
130
131      @Override
132      public boolean remove(Object o) {
133        if (o instanceof FCAObject)
134          return cxt.rowHeads().remove(((FCAObject<?, ?>) o).getIdentifier());
135        return cxt.rowHeads().remove(o);
136      }
137
138      @Override
139      public boolean containsAll(Collection<?> c) {
140        boolean b = false;
141        for (Object o : c)
142          b |= contains(o);
143        return b;
144      }
145
146      @Override
147      public boolean addAll(Collection<? extends FCAObject<M, G>> c) {
148        boolean b = false;
149        for (FCAObject<M, G> o : c)
150          b |= add(o);
151        return b;
152      }
153
154      @Override
155      public boolean retainAll(Collection<?> c) {
156        // TODO Auto-generated method stub
157        return false;
158      }
159
160      @Override
161      public boolean removeAll(Collection<?> c) {
162        boolean b = false;
163        for (Object o : c)
164          b |= remove(o);
165        return b;
166      }
167
168      @Override
169      public void clear() {
170        cxt.rowHeads().clear();
171      }
172
173      @Override
174      public int getIndexOf(FCAObject<M, G> e) {
175        return cxt.rowHeads().indexOf(e.getIdentifier());
176      }
177
178      @Override
179      public FCAObject<M, G> getElementAt(int index) throws IndexOutOfBoundsException {
180        return getObjectAtIndex(index);
181      }
182
183      @Override
184      public void changeOrder() {}
185    };
186  }
187
188  @Override
189  public boolean addObject(FCAObject<M, G> e) throws IllegalObjectException {
190    return cxt.rowHeads().add(e.getIdentifier())
191        && cxt.row(e.getIdentifier()).addAll(((FullObjectDescription<M>) e.getDescription()).getAttributes());
192  }
193
194  @Override
195  public boolean removeObject(G id) throws IllegalObjectException {
196    return cxt.rowHeads().remove(id);
197  }
198
199  @Override
200  public boolean removeObject(FCAObject<M, G> object) throws IllegalObjectException {
201    return cxt.rowHeads().remove(object.getIdentifier());
202  }
203
204  @Override
205  public boolean addAttributeToObject(M attribute, G id) throws IllegalAttributeException, IllegalObjectException {
206    return cxt.add(id, attribute);
207  }
208
209  @Override
210  public Set<M> doublePrime(Set<M> x) {
211    return cxt.intent(x);
212  }
213
214  @Override
215  public Set<FCAImplication<M>> getStemBase() {
216    return null;
217  }
218
219  @Override
220  public boolean refutes(FCAImplication<M> imp) {
221    return false;
222  }
223
224  @Override
225  public boolean isCounterExampleValid(FCAObject<M, G> counterExample, FCAImplication<M> imp) {
226    return false;
227  }
228
229  @Override
230  protected boolean followsFromBackgroundKnowledge(FCAImplication<M> implication) {
231    return false;
232  }
233
234}
