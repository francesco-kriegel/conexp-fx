025import java.util.Collections;
026import java.util.HashSet;
027import java.util.Set;
028import java.util.concurrent.ExecutionException;
029import java.util.concurrent.ExecutorService;
030import java.util.concurrent.Future;
031import java.util.function.Consumer;
032import java.util.function.Supplier;
033
034import conexp.fx.core.algorithm.nextclosures.NextClosuresState;
035import conexp.fx.core.collections.Collections3;
036import conexp.fx.core.context.Context;
037import conexp.fx.core.context.Implication;
038import conexp.fx.core.math.SetClosureOperator;
039import conexp.fx.gui.ConExpFX;
040import conexp.fx.gui.dataset.FCADataset;
041import conexp.fx.gui.exploration.HumanExpertP;
042import conexp.fx.gui.task.TimeTask;
043import conexp.fx.gui.util.Platform2;
044
045public class ParallelAttributeExploration {
046
047  public static final <M> Set<Implication<String, M>> explore(
048      final Context<String, M> cxt,
049      final Set<Implication<String, M>> backgroundKnowledge,
050      final Expert<String, M> expert,
051      final ExecutorService executor,
052      final Consumer<Implication<String, M>> implicationConsumer,
053      final Consumer<String> updateStatus,
054      final Consumer<Double> updateProgress,
055      final Supplier<Boolean> isCancelled) {
056    if (!cxt.models(backgroundKnowledge))
057      throw new IllegalArgumentException("Background knowledge does not hold in formal context");
058    final NextClosuresState<String, M, Set<M>> result = NextClosuresState.withHashSets(cxt.colHeads());
059    final int maxCardinality = cxt.colHeads().size();
060    // expert.onCancelRunnable.set(() -> result.cardinality = maxCardinality);
061    final SetClosureOperator<M> clop = backgroundKnowledge == null || backgroundKnowledge.isEmpty()
062        ? SetClosureOperator.fromImplications(result.implications, true, false)
063        : SetClosureOperator.supremum(
064            SetClosureOperator.fromImplications(result.implications, true, false),
065            SetClosureOperator.fromImplications(backgroundKnowledge, false, false));
066    for (; result.cardinality <= maxCardinality; result.cardinality++) {
067      if (isCancelled.get())
068        break;
069      final double q = ((double) result.cardinality) / ((double) maxCardinality);
070      final int p = (int) (100d * q);
071      updateStatus.accept("current cardinality: " + result.cardinality + "/" + maxCardinality + " (" + p + "%)");
072      updateProgress.accept(q);
073      final Set<Future<?>> futures = Collections3.newConcurrentHashSet();
074      result.candidates.keySet().parallelStream().filter(c -> c.size() == result.cardinality).forEach(candidate -> {
075        futures.add(executor.submit(() -> {
076          final Set<M> closure = clop.closure(candidate);
077          if (closure.size() == candidate.size()) {
078            final Set<String> candidateI = cxt.colAnd(candidate);
079            final Set<M> candidateII = cxt.rowAnd(candidateI);
080            if (candidateII.size() != candidate.size()) {
081              candidateII.removeAll(candidate);
082              final Implication<String, M> implication = new Implication<String, M>(candidate, candidateII, candidateI);
083              try {
084                Set<CounterExample<String, M>> counterExamples = Collections.emptySet();
085                do {
086                  counterExamples = expert.getCounterExamples(implication);
087                  if (!counterExamples.isEmpty())
088                    counterExamples.forEach(cex -> Platform2.runOnFXThreadAndWaitTryCatch(() -> {
089                      cex.insertIn(cxt);
090                      cex.addTo(implication);
091                    }));
092                } while (!implication.isTrivial() && !counterExamples.isEmpty());
093              } catch (InterruptedException __) {
094                result.cardinality = maxCardinality;
095              }
096              if (!implication.isTrivial()) {
097                result.implications.add(implication);
098                implicationConsumer.accept(implication);
099              }
100              final Set<M> intent = new HashSet<M>();
101              intent.addAll(implication.getPremise());
102              intent.addAll(implication.getConclusion());
103              result.addNewCandidates(intent);
104            } else
105              result.addNewCandidates(candidate);
106          } else
107            result.candidates.put(closure, result.cardinality);
108          result.candidates.remove(candidate);
109        }));
110      });
111      for (Future<?> future : futures)
112        try {
113          future.get();
114        } catch (InterruptedException | ExecutionException __) {}
115    }
116    updateStatus.accept(result.implications.size() + " implications found");
117    return result.implications;
118  }
119
120  public static final <M> TimeTask<?> createExplorationTask(FCADataset<String, M> dataset) {
121    return new TimeTask<Void>(dataset, "ParallelExploration") {
122
123      @Override
124      protected Void call() throws Exception {
125        updateProgress(0d, 1d);
126        if (isCancelled())
127          return null;
128        final HumanExpertP<M> humexp = new HumanExpertP<>(dataset.context.selection);
129        Platform2.runOnFXThreadAndWaitTryCatch(() -> {
130          humexp.init();
131          humexp.show();
132        });
133        explore(
134            dataset.context.getSelection(),
135            Collections.emptySet(),
136            humexp,
137            ConExpFX.instance.executor.tpe,
138            implication -> Platform2.runOnFXThread(() -> dataset.implications.add(implication)),
139            // dataset.implications::add,
140            status -> updateMessage(status),
141            progress -> updateProgress(progress, 1d),
142            () -> isCancelled());
143        Platform2.runOnFXThreadAndWaitTryCatch(() -> humexp.hide());
144        updateProgress(1d, 1d);
145        return null;
146      }
147    };
148  }
