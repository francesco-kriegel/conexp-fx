058  public static final <T> Set<Set<T>>
059      compute(final Set<T> baseSet, final ClosureOperator<T> clop, final boolean verbose, final ExecutorService tpe) {
060    final Map<Set<T>, Set<T>> closures = new ConcurrentHashMap<>();
061    final Set<Set<T>> candidates = Collections3.newConcurrentHashSet();
062    candidates.add(new HashSet<T>());
063    for (final AtomicInteger cardinality = new AtomicInteger(0); cardinality.get() < baseSet.size(); cardinality
064        .incrementAndGet()) {
065      System.out.println("current cardinality: " + cardinality.get());
066//      final long total = candidates.parallelStream().filter(c -> c.size() == cardinality.get()).count();
067//      final AtomicInteger current = new AtomicInteger(0);
068      while (!Collections2.filter(candidates, c -> c.size() == cardinality.get()).isEmpty()) {
069        final Set<Future<?>> futures = Collections3.newConcurrentHashSet();
070        final long count = candidates.parallelStream().filter(c -> c.size() == cardinality.get()).count();
071        System.out.println("processing " + count + " candidates");
072        for (Set<T> candidate : candidates)
073          if (candidate.size() == cardinality.get()) {
074            // candidates.parallelStream().filter(c -> c.size() == cardinality.get()).forEach(candidate -> {
075            futures.add(tpe.submit(() -> {
076              final Optional<Entry<Set<T>, Set<T>>> optional = closures
077                  .entrySet()
078                  .parallelStream()
079                  .filter(e -> candidate.containsAll(e.getKey()) && e.getValue().containsAll(candidate))
080                  .findAny();
081              final Set<T> closure;
082              if (optional.isPresent())
083                closure = optional.get().getValue();
084              else {
085                closure = clop.closure(candidate);
086//              System.out.println(cardinality.get() + " : " + current.incrementAndGet() + "/" + total + " : " + closure);
087                closures.put(candidate, closure);
088                Sets.difference(baseSet, closure).parallelStream().forEach(individual -> {
089                  final Set<T> nextCandidate = Sets.newHashSet(closure);
090                  nextCandidate.add(individual);
091                  candidates.add(nextCandidate);
092                });
093              }
094            }));
095          }
096        // });
097        candidates.removeIf(c -> c.size() == cardinality.get());
098        futures.forEach(f -> {
099          try {
100            f.get();
101          } catch (Exception e) {}
102        });
103      }
104    }
105    return closures.values().parallelStream().collect(Collectors.toSet());
106  }

108  @SafeVarargs
109  public static final <G, M> Pair<Set<Concept<G, M>>, Set<Implication<G, M>>> compute(
110      final Context<G, M> cxt,
111      final ExecutorService executor,
112      final Consumer<Concept<G, M>> conceptConsumer,
113      final Consumer<Implication<G, M>> implicationConsumer,
114      final Consumer<String> updateStatus,
115      final Consumer<Double> updateProgress,
116      final Supplier<Boolean> isCancelled,
117      final Collection<Implication<G, M>>... backgroundKnowledge) {
118//    System.out.println("invalid background implications:");
119//    Collections3.union(backgroundKnowledge).stream().filter(i -> !cxt.models(i)).forEach(System.out::println);;
120//    System.out.println();
121    if (!cxt.models(Collections3.union(backgroundKnowledge)))
122      throw new RuntimeException("The background implications are not valid in the formal context.");
123    final NextClosuresState<G, M, Set<M>> result = NextClosuresState.withHashSets(cxt.colHeads());
124    final Function<Integer, ClosureOperator<M>> clop;
125    if (backgroundKnowledge.length == 0)
126      clop = i -> ClosureOperator.fromImplications(result.implications, i, true, true);
127    else
128      clop = i -> ClosureOperator.supremum(
129          ClosureOperator.fromImplications(result.implications, i, true, true),
130          ClosureOperator.fromImplications(Collections3.union(backgroundKnowledge), false, true));
131//    final ClosureOperator<M> clop = ClosureOperator.fromImplications(result.implications, true, true);
132    final int maxCardinality = cxt.colHeads().size();
133    for (; result.cardinality <= maxCardinality; result.cardinality++) {
134      try {
135        if (isCancelled.get())
136          break;
137      } catch (Exception __) {}
138      final double q = ((double) result.cardinality) / ((double) maxCardinality);
139      final int p = (int) (100d * q);
140      updateStatus.accept("current cardinality: " + result.cardinality + "/" + maxCardinality + " (" + p + "%)");
141      updateProgress.accept(q);
142      final Set<Future<?>> futures = Collections3.newConcurrentHashSet();
143      final Set<Set<M>> cc = result.getActualCandidates();
144      cc.forEach(candidate -> {
145        futures.add(executor.submit(() -> {
146          final Set<M> closure = clop.apply(result.candidates.get(candidate)).closure(candidate);
147//          if (closure.size() == candidate.size()) {
148          if (closure.equals(candidate)) {
149            final Set<G> candidateI = cxt.colAnd(candidate);
150            final Set<M> candidateII = cxt.rowAnd(candidateI);
151            if (result.isNewIntent(candidateII)) {
152              final Concept<G, M> concept = new Concept<G, M>(candidateI, new HashSet<M>(candidateII));
153              result.concepts.add(concept);
154              conceptConsumer.accept(concept);
155              result.addNewCandidates(candidateII);
156            }
157            if (candidateII.size() != candidate.size()) {
158              candidateII.removeAll(candidate);
159              final Implication<G, M> implication = new Implication<G, M>(candidate, candidateII, candidateI);
160              result.implications.add(implication);
161              implicationConsumer.accept(implication);
162            }
163          } else {
164            result.candidates.put(closure, result.cardinality);
165          }
166//          result.candidates.remove(candidate);
167        }));
168      });
169      for (Future<?> future : futures)
170        try {
171          future.get();
172        } catch (InterruptedException | ExecutionException e) {
173          e.printStackTrace();
174        }
175      result.candidates.keySet().removeAll(cc);
176    }
177    updateStatus
178        .accept(result.concepts.size() + " concepts, and " + result.implications.size() + " implications found");
179    return result.getResultAndDispose();
180  }

190  @SafeVarargs
191  public static final <G, M> Pair<Set<Concept<G, M>>, Set<Implication<G, M>>>
192      compute(final Context<G, M> cxt, final int cores, final Collection<Implication<G, M>>... backgroundKnowledge) {
193    if (cores > Runtime.getRuntime().availableProcessors())
194      throw new IllegalArgumentException(
195          "Requested pool size is too large. VM has only " + Runtime.getRuntime().availableProcessors()
196              + " available cpus, thus a thread pool with " + cores + " cores cannot be used here.");
197//    final ThreadPoolExecutor tpe =
198//        new ThreadPoolExecutor(cores, cores, 1000, TimeUnit.MILLISECONDS, new LinkedBlockingQueue<Runnable>());
199//    tpe.prestartAllCoreThreads();
200    final ExecutorService tpe = Executors.newWorkStealingPool(cores);
201    final Pair<Set<Concept<G, M>>, Set<Implication<G, M>>> result = compute(cxt, tpe, backgroundKnowledge);
202//    tpe.purge();
203    tpe.shutdown();
204    return result;
205  }

213  public static final <G, M> TimeTask<?> createTask(FCADataset<G, M> dataset) {
214    return new TimeTask<Void>(dataset, "NextClosures") {
215
216      @Override
217      protected Void call() throws Exception {
218        updateProgress(0d, 1d);
219        if (isCancelled())
220          return null;
221        compute(
222            dataset.context.getSelection(),
223            ConExpFX.instance.executor.tpe,
224            concept -> Platform2.runOnFXThread(() -> dataset.concepts.add(concept)),
225            implication -> Platform2.runOnFXThread(() -> dataset.implications.add(implication)),
226            // dataset.concepts::add,
227            // dataset.implications::add,
228            status -> updateMessage(status),
229            progress -> updateProgress(progress, 1d),
230            () -> isCancelled());
231        updateProgress(1d, 1d);
232        return null;
233      }
234    };
235  }
