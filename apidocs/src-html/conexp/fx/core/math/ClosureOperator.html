046  @Override
047  public default Set<M> apply(final Set<M> set) {
048    return closure(set);

052
053  public default boolean close(final Set<M> set) {
054    return !set.addAll(closure(set));
055  }

058//    return set.containsAll(closure(set));
059    return set.size() == closure(set).size();

067  public static <T> ClosureOperator<T> infimum(final Iterable<ClosureOperator<T>> closureOperators) {
068    return set -> {
069      final Set<T> closure = new HashSet<T>();
070      final Iterator<ClosureOperator<T>> iterator = closureOperators.iterator();
071      if (iterator.hasNext())
072        set.addAll(iterator.next().closure(set));
073      while (iterator.hasNext())
074        set.retainAll(iterator.next().closure(set));
075      return closure;
076    };
077  }

084  public static <T> ClosureOperator<T> supremum(final Iterable<ClosureOperator<T>> closureOperators) {
085    return set -> {
086      final Set<T> closure = new HashSet<T>();
087      closure.addAll(set);
088      boolean changed = true;
089      while (changed) {
090        changed = false;
091        for (ClosureOperator<T> clop : closureOperators)
092          changed |= closure.addAll(clop.closure(closure));
093      }
094      return closure;
095    };

107  public static <G, M, C extends Set<M>> C implicativeClosure(
108      final Collection<Implication<G, M>> implications,
109      final int firstPremiseSize,
110      final boolean includePseudoClosures,
111      final boolean parallel,
112      final boolean bySize,
113      final Function<Set<M>, C> supplier,
114      final Set<M> set) {
115    final C closure = supplier.apply(set);// new HashSet<M>(set);
116    final AtomicBoolean changed = new AtomicBoolean(true);
117    final Predicate<Implication<G, M>> p;
118    if (includePseudoClosures)
119      p = i -> closure.size() > i.getPremise().size() && closure.containsAll(i.getPremise());
120    else
121      p = i -> closure.size() >= i.getPremise().size() && closure.containsAll(i.getPremise());
122    // && (closure.size() < i.getConclusion().size() || !closure.containsAll(i.getConclusion())))
123    if (bySize) {
124      int size;
125      if (firstPremiseSize > 0) {
126        size = closure.size();
127        (parallel ? implications.parallelStream() : implications.stream())
128            .filter(i -> firstPremiseSize <= i.getPremise().size())
129            .filter(p)
130            .sequential()
131            .forEach(i -> closure.addAll(i.getConclusion()));
132        changed.set(closure.size() != size);
133      }
134      while (changed.get()) {
135        size = closure.size();
136        changed.set(false);
137        (parallel ? implications.parallelStream() : implications.stream())
138            .filter(p)
139            .sequential()
140            .forEach(i -> closure.addAll(i.getConclusion()));
141        changed.set(closure.size() != size);
142      }
143    } else {
144      if (firstPremiseSize > 0) {
145        (parallel ? implications.parallelStream() : implications.stream())
146            .filter(i -> firstPremiseSize <= i.getPremise().size())
147            .filter(p)
148            .sequential()
149            .forEach(i -> changed.set(closure.addAll(i.getConclusion()) || changed.get()));
150      }
151      while (changed.get()) {
152        changed.set(false);
153        (parallel ? implications.parallelStream() : implications.stream())
154            .filter(p)
155            .sequential()
156            .forEach(i -> changed.set(closure.addAll(i.getConclusion()) || changed.get()));
157      }
158    }
159    return closure;
160  }

162  public static <G, M, C extends Set<M>> ClosureOperator<M> fromImplications(
163      final Collection<Implication<G, M>> implications,
164      final int firstPremiseSize,
165      final boolean includePseudoClosures,
166      final boolean parallel,
167      final boolean bySize,
168      final Function<Set<M>, C> supplier) {
169    return set -> implicativeClosure(
170        implications,
171        firstPremiseSize,
172        includePseudoClosures,
173        parallel,

175        supplier,
176        set);

198  public static <G, M> ClosureOperator<M>
199      fromImplicationSetLinClosure(final Collection<Implication<G, M>> implications) {
200    return set -> {
201
202      // Initialization
203      final Map<Implication<G, M>, Integer> count = new HashMap<Implication<G, M>, Integer>();
204      final Map<M, Set<Implication<G, M>>> list = new HashMap<M, Set<Implication<G, M>>>();
205      for (M attribute : set)
206        list.put(attribute, new HashSet<Implication<G, M>>());
207      for (Implication<G, M> implication : implications) {
208        for (M attribute : implication.getPremise())
209          list.put(attribute, new HashSet<Implication<G, M>>());
210        for (M attribute : implication.getConclusion())
211          list.put(attribute, new HashSet<Implication<G, M>>());
212      }
213      // final Multimap<M, Implication<G, M>> list = HashMultimap.create();
214      final Set<M> newdep = new HashSet<M>();
215      final Set<M> update = new HashSet<M>();
216      for (Implication<G, M> implication : implications) {
217        count.put(implication, implication.getPremise().size());
218        if (implication.getPremise().isEmpty())
219          newdep.addAll(implication.getConclusion());
220        for (M attribute : implication.getPremise()) {
221          if (!list.containsKey(attribute))
222            list.put(attribute, new HashSet<Implication<G, M>>());
223          list.get(attribute).add(implication);
224          // list.put(
225          // attribute,
226          // implication);
227        }
228      }
229      newdep.addAll(set);
230      update.addAll(set);
231
232      // Computation
233      while (!update.isEmpty()) {
234        final M attribute = update.iterator().next();
235        update.remove(attribute);
236        for (Implication<G, M> implication : list.get(attribute)) {
237          final Integer previous = count.get(implication);
238          count.put(implication, previous - 1);
239          if (count.get(implication) == 0) {
240            final Set<M> add = new HashSet<M>();
241            add.addAll(implication.getConclusion());
242            add.removeAll(newdep);
243            newdep.addAll(add);
244            update.addAll(add);
245          }
246        }
247      }
248
249      return newdep;
250    };
251  }

287  /**
288   * @param maxCard
289   * @param baseSet
290   * @return a closure operator, whose closed sets are those that do not exceed the given maximal cardinality, or
291   *         contains all elements from the base set.
292   */
293  public static <G, M> ClosureOperator<M> byMaximalCardinality(final int maxCard, final Collection<M> baseSet) {
294    return new ClosureOperator<M>() {
295
296      @Override
297      public boolean isClosed(final Set<M> set) {
298        return set.size() <= maxCard || set.containsAll(baseSet);
299      }
300
301      @Override
302      public boolean close(final Set<M> set) {
303        if (isClosed(set))
304          return true;
305        set.addAll(baseSet);
306        return false;
307      }
308
309      @Override
310      public Set<M> closure(final Set<M> set) {
311        if (isClosed(set))
312          return new HashSet<M>(set);
313        return new HashSet<M>(baseSet);
314      }
315
316    };
317  }

320   * @param minSupp
321   * @param cxt
322   * @return a closure operator, whose closed sets are those that have an extent with at least minSupp elements in the
323   *         given formal context cxt, or contain all elements from the context's codomain.
324   */
325  public static <G, M> ClosureOperator<M> byMinimalSupport(final int minSupp, final Context<G, M> cxt) {
326    return new ClosureOperator<M>() {
327
328      @Override
329      public boolean isClosed(final Set<M> set) {
330        return cxt.colAnd(set).size() >= minSupp || set.containsAll(cxt.colHeads());
331      }
332
333      @Override
334      public boolean close(final Set<M> set) {
335        if (isClosed(set))
336          return true;
337        set.addAll(cxt.colHeads());
338        return false;
339      }
340
341      @Override
342      public Set<M> closure(final Set<M> set) {
343        if (isClosed(set))
344          return new HashSet<M>(set);
345        return new HashSet<M>(cxt.colHeads());
346      }
347
348    };

362      }
363
364      @Override

366        if (isClosed(set))
367          return true;
368        set.addAll(elements);
369        return false;
370      }

372      @Override
373      public Set<M> closure(final Set<M> set) {
374        if (isClosed(set))
375          return new HashSet<M>(set);

436  public static <M> ClosureOperator<M> top(final Set<M> baseSet) {
437    return new ClosureOperator<M>() {
438
439      @Override
440      public final boolean isClosed(final Set<M> set) {
441        return set.containsAll(baseSet) && baseSet.containsAll(set);
442      }
443
444      @Override
445      public final boolean close(final Set<M> set) {
446        set.retainAll(baseSet);
447        return !set.addAll(baseSet);
448      }
449
450      @Override
451      public final Set<M> closure(final Set<M> set) {
452        return new HashSet<M>(baseSet);
453      }
